import { Prop, Schema, SchemaFactory } from "@nestjs/mongoose";
import mongoose, { SchemaTypes, Types } from "mongoose";
import { Media } from "../../../modules/car/schema/car.schema";

// relationship
import { Store } from "../../store/schema/store.schema";

/**
 * Represents an Award in the system.
 * 
 * @remarks
 * This class is decorated with `@Schema` to define it as a Mongoose schema with timestamps and no version key.
 */
@Schema({ timestamps: true, versionKey: false })
export class Award {
    /**
     * The unique identifier for the award.
     * @optional
     */
    _id?: mongoose.Types.ObjectId;
    
    /**
     * The name of the award.
     * @required
     */
    @Prop({ type: String, required: true })
    name: string;

    /**
     * A description of the award.
     * @required
     */
    @Prop({ type: String, required: true })
    description: string;

    /**
     * The image associated with the award.
     * @required
     */
    @Prop({ type: SchemaTypes.Mixed, required: true })
    image: Media;

    /**
     * The stores associated with this award.
     * @optional
     */
    @Prop({ type: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Store' }] })
    store?: Store[] | Types.ObjectId[];

    /**
     * Indicates whether the award is active.
     * @default true
     * @required
     */
    @Prop({ type: Boolean, required: true, default: true })
    active: boolean;

    /**
     * The identifier of the user who created the award.
     * @required
     */
    @Prop({ type: String, required: true })
    createdBy: Types.ObjectId;
}

export const AwardSchema = SchemaFactory.createForClass(Award);
