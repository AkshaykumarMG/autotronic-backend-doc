import { Types } from "mongoose";
import { User } from "../../user/schema/user.schema";
import { ResultStatus, ResultVehicleFinanceApplication, VahicleCategory, VehicleFinanceApplication } from "../schema/vehicle-finance-application.schema";
import { IsArray, IsBoolean, IsDefined, IsEnum, IsMongoId, IsNumber, IsObject, IsOptional, IsString } from "class-validator";
import { ApiProperty, ApiPropertyOptional } from "@nestjs/swagger";

/**
 * Data Transfer Object (DTO) representing the result of a vehicle finance application.
 * This class implements the ResultVehicleFinanceApplication interface.
 */
export class ResultVehicleFinanceApplicationDto implements ResultVehicleFinanceApplication {
    /**
     * The unique identifier for the finance application result.
     * @optional
     */
    @IsMongoId()
    @IsOptional()
    _id?: Types.ObjectId;

    /**
     * The name of the lender providing the finance offer.
     */
    @IsString()
    @ApiProperty()
    lender: string;
    
    /**
     * The credit score or internal score used for the application.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    score: number;

    /**
     * The current status of the finance application result.
     */
    @IsEnum(ResultStatus)
    @ApiProperty({ enum: ResultStatus })
    status: ResultStatus;

    /**
     * The term of the finance offer in months.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    term: number;

    /**
     * The total amount of the finance offer.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    amount: number;

    /**
     * The monthly payment amount for the finance offer.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    monthly_payment: number;

    /**
     * The Annual Percentage Rate (APR) for the finance offer.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    apr: number;

    /**
     * Additional notes or comments related to the finance application result.
     */
    @IsArray()
    @ApiProperty({ type: [String] })
    notes: string[];
}

/**
 * Data Transfer Object (DTO) for creating a vehicle finance application.
 * This class implements the VehicleFinanceApplication interface and contains
 * all the necessary information for submitting a new finance application.
 */
export class CreateVehicleFinanceApplicationDto implements VehicleFinanceApplication {

    /**
     * The category of the vehicle.
     */
    @IsEnum(VahicleCategory)
    @ApiProperty({ enum: VahicleCategory })
    category: VahicleCategory;

    /**
     * The make (manufacturer) of the vehicle.
     */
    @IsString()
    @IsDefined()
    @ApiProperty()
    make: string;

    /**
     * The model of the vehicle.
     */
    @IsString()
    @IsDefined()
    @ApiProperty()
    model: string;

    /**
     * The manufacturing year of the vehicle.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    year: number;

    /**
     * The price of the vehicle.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    price: number;

    /**
     * The down payment amount for the vehicle.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    down_payment: number;

    /**
     * The user associated with this finance application.
     * Can be either a MongoDB ObjectId or a User object.
     */
    @IsMongoId()
    @ApiProperty({ type: String })
    user?: Types.ObjectId | User;

    /**
     * Indicates whether to save the user's preferences.
     */
    @IsBoolean()
    @IsOptional()
    @ApiPropertyOptional({ type: 'boolean' })
    save_preferance: boolean;

    /**
     * The desired monthly payment amount.
     */
    @IsNumber()
    @IsOptional()
    @ApiPropertyOptional({ type: 'number' })
    monthly_payment: number;

    /**
     * The annual income of the applicant.
     */
    @IsNumber()
    @IsOptional()
    @ApiPropertyOptional({ type: 'number' })
    annual_income: number;

    /**
     * The estimated credit score of the applicant.
     */
    @IsNumber()
    @IsOptional()
    @ApiPropertyOptional({ type: 'number' })
    estimated_credit_score: number;

    /**
     * The results of the vehicle finance application.
     * This is an array of ResultVehicleFinanceApplicationDto objects.
     */
    @IsArray()
    @IsOptional()
    // @ApiProperty({ type: ResultVehicleFinanceApplication })
    result?: ResultVehicleFinanceApplicationDto[];
}
