import { Types } from "mongoose";
import { User } from "../../user/schema/user.schema";
import { ResidencyStatus, ResultStatus, ResultVehicleFinanceApplication, SubVehicleFinanceApplication, VahicleCategory, WorkType } from "../schema/vehicle-finance-application.schema";
import { IsArray, IsBoolean, IsDefined, IsEmail, IsEnum, IsMongoId, IsNumber, IsObject, IsOptional, IsString, ValidateNested } from "class-validator";
import { Type } from "class-transformer";
import { ResultVehicleFinanceApplicationDto } from "./create-vehicle-finance-application.dto";

/**
 * Data Transfer Object for creating a user with a vehicle finance application.
 * This DTO includes details about the vehicle, user information, and financial details.
 */
export class CreateUserVehicleFinanceApplicationDto {
    /** The category of the vehicle. */
    @IsEnum(VahicleCategory)
    category: VahicleCategory;

    /** The make (brand) of the vehicle. */
    @IsString()
    @IsDefined()
    make: string;

    /** The model of the vehicle. */
    @IsString()
    @IsDefined()
    model: string;

    /** The manufacturing year of the vehicle. */
    @IsNumber()
    year: number;

    /** The price of the vehicle. */
    @IsNumber()
    price: number;

    /** The down payment amount for the vehicle. */
    @IsNumber()
    down_payment: number;

    /** The user's ID or User object (optional). */
    @IsMongoId()
    @IsOptional()
    user?: Types.ObjectId | User;

    /** The annual income of the applicant. */
    @IsNumber()
    annual_income: number;

    /** The estimated credit score of the applicant. */
    @IsNumber()
    estimated_credit_score: number;

    /** The desired monthly payment amount. */
    @IsNumber()
    monthly_payment: number;
    
    /** The username of the applicant. */
    @IsString()
    username: string;
    
    /** Whether to save the applicant's preferences (optional). */
    @IsBoolean()
    @IsOptional()
    save_preferance: boolean;

    /** The email address of the applicant. */
    @IsString()
    @IsEmail()
    email: string;

    /** The password for the user account (optional). */
    @IsString()
    @IsOptional()
    password?: string;

    /** The phone number of the applicant. */
    @IsNumber()
    phone: number;
}

/**
 * Data Transfer Object for creating a result of a vehicle finance application.
 * This DTO includes details about the lending offer and application status.
 */
export class CreateResultVehicleFinanceApplicationDto implements ResultVehicleFinanceApplication {
    /** The name of the lender providing the offer. */
    @IsString()
    lender: string;

    /** The score assigned to this finance application result. */
    @IsNumber()
    score: number;

    /** The status of the finance application result. */
    @IsEnum(ResultStatus)
    status: ResultStatus;

    /** The monthly payment amount for this finance offer. */
    @IsNumber()
    monthly_payment: number;

    /** The term (duration) of the finance offer in months. */
    @IsNumber()
    term: number;

    /** The total amount of the finance offer. */
    @IsNumber()
    amount: number;

    /** The Annual Percentage Rate (APR) of the finance offer. */
    @IsNumber()
    apr: number;

    /** Additional notes or comments about the finance offer. */
    @IsArray()
    notes: string[];
}

/**
 * Data Transfer Object for creating a sub vehicle finance application.
 * This DTO includes detailed information about the applicant and their financial situation.
 */
export class CreateSubVehicleFinanceApplicationDto implements SubVehicleFinanceApplication {
    /** The unique identifier for the sub application. Optional MongoDB ObjectId. */
    @IsMongoId()
    @IsOptional()
    _id?: Types.ObjectId;

    /** The age of the applicant. */
    @IsNumber()
    age: number;
    
    /** The annual income of the applicant. */
    @IsNumber()
    annual_income: number;

    /** Indicates whether the applicant owns a house. */
    @IsBoolean()
    house_status: boolean;
    
    /** Indicates whether the applicant has an existing car loan. */
    @IsBoolean()
    car_loan_status: boolean;
    
    /** Indicates whether the applicant has filed for bankruptcy. */
    @IsBoolean()
    bankrupt: boolean;
    
    /** The type of work or employment status of the applicant. */
    @IsEnum(WorkType)
    work_type: WorkType;
    
    /** The residency status of the applicant. */
    @IsEnum(ResidencyStatus)
    residency_status: ResidencyStatus;
    
    /** The down payment amount for the vehicle. */
    @IsNumber()
    down_payment: number;
    
    /** The estimated credit score of the applicant. */
    @IsNumber()
    estimated_credit_score: number;
    
    /** The desired monthly payment amount. */
    @IsNumber()
    monthly_payment: number;

    /** An array of finance application results. Optional. */
    @IsArray()
    @Type(() => ResultVehicleFinanceApplicationDto)
    @ValidateNested({ each: true })
    @IsOptional()
    result: ResultVehicleFinanceApplicationDto[];
}
