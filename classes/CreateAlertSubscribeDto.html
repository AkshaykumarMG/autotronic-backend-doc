import { Types } from "mongoose";
import { User } from "../../../modules/user/schema/user.schema";
import { AlertSubscribe } from "../schema/alert-subscribe.schema";
import { IsArray, IsBoolean, IsEmail, IsMongoId, IsObject, IsOptional, IsString } from "class-validator";
import { ApiProperty } from "@nestjs/swagger";

/**
 * Data Transfer Object for creating an alert subscription.
 * This DTO implements the AlertSubscribe interface and defines the structure
 * for creating a new alert subscription in the system.
 */
export class CreateAlertSubscribeDto implements AlertSubscribe {
    /**
     * The email address of the subscriber.
     * This field is validated to ensure it's a valid email format.
     */
    @IsEmail()
    @IsString()
    @ApiProperty()
    email: string;

    /**
     * The postal code associated with the subscription.
     */
    @IsString()
    @ApiProperty()
    postal_code: string;

    /**
     * Additional details about the subscription.
     * This is an optional field that can contain any object.
     */
    @IsObject()
    @IsOptional()
    @ApiProperty()
    detail: Object;

    /**
     * An array of filter objects defining the criteria for the alert.
     */
    @IsArray()
    @ApiProperty({ type: Array<Object> })
    filter: Array<Object>;

    /**
     * The user associated with this subscription.
     * This is an optional field that can be either a User object or a MongoDB ObjectId.
     */
    @IsString()
    @IsMongoId()
    @IsOptional()
    @ApiProperty({ type: String })
    user?: User | Types.ObjectId;

    /**
     * Indicates whether the subscription is active.
     * This is an optional boolean field.
     */
    @IsBoolean()
    @IsOptional()
    @ApiProperty()
    active: boolean;
}
