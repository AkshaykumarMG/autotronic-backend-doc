import { ApiProperty } from "@nestjs/swagger";
import { IsArray, IsDefined, IsNumber, IsNumberString, IsOptional, IsString } from "class-validator";

/**
 * Represents a data transfer object (DTO) for creating user filter criteria.
 * This class is used to validate and structure the incoming request data for filtering car data.
 */
export class CreateUserFilterDto {

    @IsDefined()
    @IsNumber()
    @ApiProperty()
    lat: number;
  
    @IsDefined()
    @IsNumber()
    @ApiProperty()
    lng: number;

    @IsDefined()
    @IsNumberString()
    @ApiProperty()
    distance: number;

    @IsDefined()
    @IsString()
    @IsOptional()
    @ApiProperty()
    make: string;

    @IsDefined()
    @IsString()
    @IsOptional()
    @ApiProperty()
    model: string;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    trim: Array<any>;
    
    @IsArray()
    @IsOptional()
    @ApiProperty()
    interior_color: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    year: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    asking_price: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    mileage: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    transmission: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    body_type: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    condition: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    seller_type: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    exterior_color: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    drivetrain: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    fuel_type: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    engine: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    seating_capacity: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    doors: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    features: Array<any>;

    @IsArray()
    @IsOptional()
    @ApiProperty()
    home_services: Array<any>;
}
