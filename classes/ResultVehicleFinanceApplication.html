import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import mongoose, { HydratedDocument, SchemaTypes, Types } from 'mongoose';
import { User } from '../../user/schema/user.schema';

export type VehicleFinanceApplicationDocument = HydratedDocument<VehicleFinanceApplication>;

export enum VahicleCategory {
    car = 'car',
    truck = 'truck',
    bus = 'bus',
    'commercial_vehicals' = 'commercial_vehicals'
}

export enum WorkType {
    contract = 'contract',
    full_time = 'full_time',
    part_time = 'part_time',
    retired = 'retired',
    student = 'student'
}

export enum ResidencyStatus {
    canadian_citizen = 'canadian_citizen',
    permenent_resident = 'permenent_resident',
    other = 'other'
}

export enum ResultStatus {
    Approved = 'Approved',
    'Not Approved' = 'Not Approved'
}

/**
 * Represents the result of a vehicle finance application.
 * This class contains details about the financing offer from a lender.
 */
@Schema({ timestamps: false, versionKey: false })
export class ResultVehicleFinanceApplication {
    /** The name of the lender providing the financing offer. */
    @Prop({ type: String, required: true })
    lender: string;

    /** The score assigned to this financing offer. Higher scores typically indicate better offers. */
    @Prop({ type: Number, required: true })
    score: number;

    /** The status of the application result, either 'Approved' or 'Not Approved'. */
    @Prop({ type: String, required: true, enum: ResultStatus })
    status: ResultStatus;

    /** The term of the loan in months. */
    @Prop({ type: Number, required: true })
    term: number;

    /** The total amount of the loan offered. */
    @Prop({ type: Number, required: true })
    amount: number;

    /** The monthly payment amount for the loan. */
    @Prop({ type: Number, required: true })
    monthly_payment: number;
    
    /** The Annual Percentage Rate (APR) of the loan. */
    @Prop({ type: Number, required: true })
    apr: number;
    
    /** Additional notes or comments about the financing offer. */
    @Prop({ type: [String], default: [] })
    notes: string[];
}
export const ResultVehicleFinanceApplicationSchema = SchemaFactory.createForClass(ResultVehicleFinanceApplication);


/**
 * Represents a sub-application for vehicle financing.
 * This class contains detailed information about the applicant and their financial situation.
 */
@Schema({ timestamps: true, versionKey: false })
export class SubVehicleFinanceApplication {
    /** The age of the applicant in years. */
    @Prop({ type: Number, required: true })
    age: number;

    /** The annual income of the applicant in the applicable currency. */
    @Prop({ type: Number, required: true })
    annual_income: number;

    /** Indicates whether the applicant owns a house. Default is false. */
    @Prop({ type: Boolean, default: false })
    house_status: boolean;
    
    /** Indicates whether the applicant has an existing car loan. Default is false. */
    @Prop({ type: Boolean, default: false })
    car_loan_status: boolean;
    
    /** Indicates whether the applicant has been bankrupt. Default is false. */
    @Prop({ type: Boolean, default: false })
    bankrupt: boolean;
    
    /** The type of work the applicant is engaged in. Must be one of the defined WorkType enum values. */
    @Prop({ type: String, required: true, enum: WorkType })
    work_type: WorkType;
    
    /** The residency status of the applicant. Must be one of the defined ResidencyStatus enum values. */
    @Prop({ type: String, required: true, enum: ResidencyStatus })
    residency_status: ResidencyStatus;
    
    /** The amount of down payment the applicant is willing to make. Default is 0. */
    @Prop({ type: Number, default: 0 })
    down_payment: number;
    
    /** The estimated credit score of the applicant. Default is 0. */
    @Prop({ type: Number, default: 0 })
    estimated_credit_score: number;

    /** The monthly payment amount the applicant is willing to make. Default is 0. */
    @Prop({ type: Number, default: 0 })
    monthly_payment: number;

    /** An optional array of financing results for this sub-application. */
    @Prop({ type: [ResultVehicleFinanceApplicationSchema] })
    result?: ResultVehicleFinanceApplication[];
}
export const SubVehicleFinanceApplicationSchema = SchemaFactory.createForClass(SubVehicleFinanceApplication);

/**
 * Represents a vehicle finance application.
 * This class contains detailed information about the vehicle, applicant, and financing details.
 */
@Schema({ timestamps: true, versionKey: false })
export class VehicleFinanceApplication {
    /** The unique identifier for the application. */
    _id?: mongoose.Types.ObjectId;

    /** The category of the vehicle. Must be one of the defined VahicleCategory enum values. */
    @Prop({ type: String, required: true, index: true, enum: VahicleCategory })
    category: VahicleCategory;

    /** The make (manufacturer) of the vehicle. */
    @Prop({ type: String, required: true, index: true })
    make: string;

    /** The model of the vehicle. */
    @Prop({ type: String, required: true, index: true })
    model: string;

    /** The manufacturing year of the vehicle. */
    @Prop({ type: Number, required: true, index: true })
    year: number;

    /** The price of the vehicle. */
    @Prop({ type: Number, required: true })
    price: number;

    /** The amount of down payment for the vehicle. Defaults to 0. */
    @Prop({ type: Number, default: 0 })
    down_payment: number;

    /** The user associated with this application. Can be a User object or ObjectId. */
    @Prop({ type: mongoose.Schema.Types.ObjectId, ref: User.name })
    user?: User | Types.ObjectId;

    /** The annual income of the applicant. */
    @Prop({ type: Number, required: true })
    annual_income: number;

    /** The estimated credit score of the applicant. */
    @Prop({ type: Number, required: true })
    estimated_credit_score: number;

    /** The age of the applicant. Optional. */
    @Prop({ type: Number })
    age?: number;

    /** Indicates whether the applicant owns a house. Optional. */
    @Prop({ type: Boolean })
    house_status?: boolean;

    /** Indicates whether to save the applicant's preferences. */
    @Prop({ type: Boolean })
    save_preferance: boolean;

    /** Indicates whether the applicant has an existing car loan. Optional. */
    @Prop({ type: Boolean })
    car_loan_status?: boolean;

    /** Indicates whether the applicant has been bankrupt. Optional. */
    @Prop({ type: Boolean })
    bankrupt?: boolean;

    /** The type of work the applicant is engaged in. Must be one of the defined WorkType enum values. Optional. */
    @Prop({ type: String, enum: WorkType })
    work_type?: WorkType;

    /** The residency status of the applicant. Must be one of the defined ResidencyStatus enum values. Optional. */
    @Prop({ type: String, enum: ResidencyStatus })
    residency_status?: ResidencyStatus;

    /** The monthly payment amount the applicant is willing to make. Defaults to 0. Optional. */
    @Prop({ type: Number, default: 0, min: 0 })
    monthly_payment?: number;

    /** An array of sub-applications for this vehicle finance application. Optional. */
    @Prop({ type: [SubVehicleFinanceApplicationSchema] })
    data?: SubVehicleFinanceApplication[];

    /** An array of financing results for this application. Optional. */
    @Prop({ type: [ResultVehicleFinanceApplicationSchema] })
    result?: ResultVehicleFinanceApplication[];
}
export const VehicleFinanceApplicationSchema = SchemaFactory.createForClass(VehicleFinanceApplication);
