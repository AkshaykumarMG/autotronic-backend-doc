import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import mongoose, { HydratedDocument, Types } from 'mongoose';

// relationship
import { Car } from '../../../modules/car/schema/car.schema';
import { Store } from '../../../modules/store/schema/store.schema';
import { User } from '../../../modules/user/schema/user.schema';

export type InquiryDocument = HydratedDocument<Inquiry>;

export enum InquiryType {
    contact = 'contact',
    appointment = 'appointment'
}

export enum InquiryStatus {
    new = 'new',
    pending = 'pending',
    done = 'done'
}

export interface Timeline {
    status: InquiryStatus,
    note?: string,
    user: string
}

const TimelineSchme = new mongoose.Schema({
    note: { type: String, require: true },
    user: { type: mongoose.Schema.Types.ObjectId, ref: 'User', require: true },
    status: { type: String, enum: InquiryStatus, require: true },
    duration: { type: Date, default: new Date() },
}, { _id: false });

/**
 * Represents an inquiry in the system.
 * 
 * This class defines the structure and properties of an inquiry,
 * including references to related entities, inquiry details, and status information.
 */
@Schema({ timestamps: true, versionKey: false })
export class Inquiry {
    /** The unique identifier for the inquiry. */
    _id?: mongoose.Types.ObjectId;

    /** The car associated with this inquiry. Required. */
    @Prop({ type: mongoose.Schema.Types.ObjectId, ref: Car.name, required: true })
    car: Car | Types.ObjectId;

    /** The store associated with this inquiry, if any. Indexed for faster queries. */
    @Prop({ type: mongoose.Schema.Types.ObjectId, ref: Store.name, index: true })
    store?: Store | Types.ObjectId;

    /** The user associated with this inquiry, if any. Indexed for faster queries. */
    @Prop({ type: mongoose.Schema.Types.ObjectId, ref: User.name, index: true })
    user?: User | Types.ObjectId;

    /** The type of inquiry. Defaults to 'contact'. */
    @Prop({ type: String, enum: InquiryType, default: InquiryType.contact })
    type: InquiryType;

    /** The timeline of the inquiry, tracking status changes and notes. */
    @Prop({ type: [TimelineSchme] })
    timeline?: Timeline[];

    /** The username of the person making the inquiry. Required. */
    @Prop({ type: String, required: true })
    username?: string;

    /** The email address of the person making the inquiry. Required. */
    @Prop({ type: String, required: true })
    email: string;

    /** The phone number of the person making the inquiry. Required. */
    @Prop({ type: Number, required: true })
    phone: number;

    /** The message content of the inquiry. Required. */
    @Prop({ type: String, required: true })
    message: string;

    /** The date associated with the inquiry. */
    @Prop({ type: Date })
    date: Date;

    /** The postal code associated with the inquiry. */
    @Prop({ type: String })
    postal_code: string;

    /** Indicates if the inquiry is active. Defaults to true. */
    @Prop({ type: Boolean, default: true })
    active?: boolean;

    /** The current status of the inquiry. Defaults to 'new'. */
    @Prop({ type: String, enum: InquiryStatus, default: InquiryStatus.new })
    status?: InquiryStatus;

    /** Additional notes for the inquiry. */
    @Prop({ type: String })
    note?: string;
}


export const InquirySchema = SchemaFactory.createForClass(Inquiry);

InquirySchema.index({ location: "2dsphere" });
