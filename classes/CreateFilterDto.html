import { Schema } from "mongoose";
import { Type } from "class-transformer";
import { IsNumberOrString } from "../../../utilities/other/is-number-or-string.validator";
import { DataElement, Filter, FilterDataType, FilterOperation, FilterType, labelValue, labelValueJoin } from "../schema/filter.schema";
import { ArrayMinSize, IsArray, IsBoolean, IsDefined, IsEnum, IsMongoId, IsNotEmpty, IsNotEmptyObject, IsNumber, IsObject, IsOptional, IsString, MinLength, Validate, ValidateIf, ValidateNested, isDefined } from "class-validator";
import { ApiExtraModels, ApiProperty, getSchemaPath } from "@nestjs/swagger";

enum VersionEnum {
    lable_value_join = 'lvj',
    lable_value = 'lv',
    data_element = 'de',
}

/**
 * Represents a discriminator for version-based type selection.
 */
class Discriminator {
    /**
     * The version of the data structure.
     */
    @IsEnum(VersionEnum)
    version: VersionEnum;
}

/**
 * Represents a label-value pair with an optional count.
 * Implements the labelValue interface.
 */
class LabelValueDTO implements labelValue {
    /**
     * The label of the item.
     */
    @IsString()
    @IsNotEmpty()
    @MinLength(1)
    @ApiProperty()
    label: string;

    /**
     * The value of the item, which can be either a string or a number.
     */
    @IsNotEmpty()
    @Validate(IsNumberOrString)
    @ApiProperty()
    value: string | number;

    /**
     * An optional count associated with the label-value pair.
     */
    @IsOptional()
    count: number;
}

/**
 * Represents a data element with various properties.
 * Implements the DataElement interface.
 */
class DataElementDTO implements DataElement {
    /**
     * The name of the data element.
     */
    @IsString()
    @IsNotEmpty()
    @MinLength(4)
    @ApiProperty()
    name: string;

    /**
     * The type of the filter data.
     */
    @IsNotEmpty()
    @IsEnum(FilterDataType)
    @ApiProperty({ enum: FilterDataType })
    type: FilterDataType;

    /**
     * The placeholder text for textbox or dropdown types.
     */
    @ValidateIf(o => o.type === 'textbox' || o.type === 'dropdown')
    @IsNotEmpty()
    @ApiProperty()
    placeholder: string;

    /**
     * An array of label-value pairs for dropdown type.
     */
    @ValidateIf(o => o.type === 'dropdown')
    @IsArray()
    @ArrayMinSize(1)
    @ValidateNested({ each: true })
    @Type(() => LabelValueDTO)
    @ApiProperty({ type: LabelValueDTO })
    data: LabelValueDTO[];
}


/**
 * Represents a label-value pair with an additional join property and an optional count.
 * Implements the labelValueJoin interface.
 */
class LabelValueJoinDTO implements labelValueJoin {
    /**
     * The join property, representing a connection or relationship.
     */
    @IsString()
    @IsNotEmpty()
    @ApiProperty()
    join: string;

    /**
     * The label of the item.
     */
    @IsString()
    @IsNotEmpty()
    @MinLength(1)
    @ApiProperty()
    label: string;

    /**
     * The value of the item, which can be either a string or a number.
     */
    @IsNotEmpty()
    @Validate(IsNumberOrString)
    @ApiProperty()
    value: string | number;

    /**
     * An optional count associated with the label-value pair.
     */
    @IsOptional()
    count: number;
}

@ApiExtraModels(LabelValueDTO, LabelValueJoinDTO, DataElementDTO)
export class CreateFilterDto implements Filter {
    @IsDefined()
    @IsNotEmptyObject()
    @IsObject()
    @ValidateNested({ each: true })
    @Type(() => LabelValueDTO)
    @ApiProperty({ type: LabelValueDTO })
    name: LabelValueDTO;

    @IsDefined()
    @IsEnum(FilterType)
    @ApiProperty({ enum: FilterType })
    type: FilterType;

    @IsOptional()
    @IsEnum(FilterOperation)
    @ApiProperty({ enum: FilterOperation })
    operation: FilterOperation;

    @IsBoolean()
    @ApiProperty({ type: Boolean })
    button: boolean;

    // @IsDefined()
    // @IsBoolean()
    // default_hide: Boolean;

    @IsNumber()
    @IsDefined()
    @ApiProperty({ type: Number })
    position: number;

    @IsDefined()
    @IsBoolean()
    @ApiProperty({ type: Boolean })
    dependant: boolean;

    @ValidateIf(o => o.dependant === true)
    @IsNotEmpty()
    @IsString()
    @ApiProperty({ type: String })
    dependant_field: string;

    @IsDefined()
    @IsArray()
    @ArrayMinSize(1)
    @ValidateNested({ each: true })
    @Type(() => Discriminator, {
        discriminator: {
            property: 'version',
            subTypes: [
                { value: LabelValueDTO, name: VersionEnum.lable_value },
                { value: LabelValueJoinDTO, name: VersionEnum.lable_value_join },
                { value: DataElementDTO, name: VersionEnum.data_element },
            ],
        },
        keepDiscriminatorProperty: true,
    })
    @ApiProperty({
        type: 'array',
        oneOf: [
            { $ref: getSchemaPath(LabelValueDTO) },
            { $ref: getSchemaPath(LabelValueJoinDTO) },
            { $ref: getSchemaPath(DataElementDTO) }
        ],
    })
    data: DataElement[] | labelValue[] | labelValueJoin[];

    @IsOptional()
    @IsMongoId()
    createdBy: Schema.Types.ObjectId;
}
