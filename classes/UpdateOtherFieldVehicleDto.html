import { Types } from "mongoose";
import { ResidencyStatus, ResultVehicleFinanceApplication, WorkType } from "../schema/vehicle-finance-application.schema";
import { IsBoolean, IsEnum, IsMongoId, IsNumber, IsObject, IsOptional } from "class-validator";
import { ApiProperty, ApiPropertyOptional } from "@nestjs/swagger";

/**
 * Data Transfer Object (DTO) for updating other fields of a vehicle finance application.
 * This class defines the structure for updating various details related to a vehicle finance application.
 */
export class UpdateOtherFieldVehicleDto {
    /**
     * The unique identifier of the user associated with this application.
     */
    @IsMongoId()
    @ApiProperty({ type: 'string' })
    user: Types.ObjectId;

    /**
     * The age of the applicant.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    age: number;

    /**
     * The type of work or employment of the applicant.
     */
    @IsEnum(WorkType)
    @ApiProperty({ enum: WorkType })
    work_type: WorkType;
    
    /**
     * The residency status of the applicant.
     */
    @IsEnum(ResidencyStatus)
    @ApiProperty({ enum: ResidencyStatus })
    residency_status: ResidencyStatus;
    
    /**
     * The annual income of the applicant.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    annual_income: number;

    /**
     * The estimated credit score of the applicant.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    estimated_credit_score: number;
    
    /**
     * The down payment amount for the vehicle.
     */
    @IsNumber()
    @ApiProperty({ type: 'number' })
    down_payment: number;

    /**
     * The optional monthly payment amount for the vehicle financing.
     */
    @IsNumber()
    @IsOptional()
    @ApiPropertyOptional({ type: 'number' })
    monthly_payment: number;

    /**
     * Indicates whether the applicant owns a house.
     */
    @IsBoolean()
    @ApiProperty({ type: 'boolean' })
    house_status: boolean;

    /**
     * Indicates whether the applicant has an existing car loan.
     */
    @IsBoolean()
    @ApiProperty({ type: 'boolean' })
    car_loan_status: boolean;
    
    /**
     * Indicates whether the applicant has been declared bankrupt.
     */
    @IsBoolean()
    @ApiProperty({ type: 'boolean' })
    bankrupt: boolean;

    /**
     * Indicates whether to save the applicant's preferences.
     */
    @IsBoolean()
    @ApiProperty({ type: 'boolean' })
    save_preferance: boolean;

    /**
     * Optional object containing the result of the vehicle finance application.
     */
    @IsObject()
    @IsOptional()
    result: ResultVehicleFinanceApplication;
}
