/**
 * Represents a data transfer object for an address.
 * This class is used to validate and transfer address data between the client and server.
 */
class AddressDTO {

    /**
     * The unit number of the address.
     * @type {number}
     */
    @IsNumber()
    @IsNotEmpty()
    @ApiProperty({ type: Number })
    unit: number;

    /**
     * The street name of the address.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @ApiProperty()
    street: string;

    /**
     * The city of the address.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @ApiProperty()
    city: string;

    /**
     * The state of the address.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @ApiProperty()
    state: string;

    /**
     * The country of the address.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @ApiProperty()
    country: string;

    /**
     * The postal code of the address.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @ApiProperty()
    postal_code: string;
}

/**
 * Data Transfer Object (DTO) for creating a new store.
 * This class implements the Store interface and includes validation decorators for each property.
 */
export class CreateStoreDto implements Store {
    /**
     * The address of the store.
     * @type {AddressDTO}
     */
    @IsObject()
    @Type(() => AddressDTO)
    @ApiProperty({ type: AddressDTO })
    address: AddressDTO;

    /**
     * The website URL of the store.
     * @type {string}
     */
    @IsString()
    @IsOptional()
    @ApiPropertyOptional()
    website: string;

    /**
     * The phone number of the store.
     * @type {number}
     */
    @IsNumber()
    @ApiProperty({ type: Number })
    phone: number;

    /**
     * The user associated with the store.
     * @type {User | Types.ObjectId}
     */
    @IsString()
    @IsOptional()
    @ApiProperty({ type: String })
    user: User | Types.ObjectId;

    /**
     * The Google review link for the store.
     * @type {string}
     */
    @IsString()
    @IsOptional()
    @ApiProperty()
    google_review: string;

    /**
     * The logo of the store.
     * @type {Media}
     */
    @IsOptional()
    @IsObject()
    @Type(() => MediaDTO)
    @ApiProperty({ type: MediaDTO })
    logo: Media;

    /**
     * The name of the store.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @IsDefined()
    @ApiProperty()
    name: string;

    /**
     * The license number of the store.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @IsDefined()
    @ApiProperty()
    license_number: string;

    /**
     * The email address of the store.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @IsDefined()
    @IsEmail()
    @ApiProperty()
    email: string;

    /**
     * The alias or alternative name of the store.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @IsDefined()
    @ApiProperty()
    alias: string;

    /**
     * The remote services offered by the store.
     * @type {RemoteService[] | Types.ObjectId[]}
     */
    @IsArray()
    @IsOptional()
    @ApiProperty({ type: String })
    remote_services: RemoteService[] | Types.ObjectId[];

    /**
     * The custom services offered by the store.
     * @type {[string]}
     */
    @IsArray()
    @IsOptional()
    @ApiProperty({ type: [String] })
    custom_services: [string];

    /**
     * The geographical location of the store.
     * @type {LocationDTO}
     */
    @IsObject()
    @Type(() => LocationDTO)
    @ApiProperty({ type: LocationDTO })
    location: LocationDTO;

    /**
     * The operating hours of the store.
     * @type {TimeDTO}
     */
    @IsObject()
    @Type(() => TimeDTO)
    @ApiProperty({ type: TimeDTO })
    time: TimeDTO;

    /**
     * The Annual Percentage Rate (APR) for financing offered by the store.
     * @type {number}
     */
    @IsNumber()
    @IsOptional()
    @ApiProperty({ type: Number })
    finance_apr: number;
