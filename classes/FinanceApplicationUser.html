@Schema({ timestamps: true, versionKey: false })
export class FinanceApplication {

    _id?: mongoose.Types.ObjectId;

    @Prop({ type: mongoose.Schema.Types.ObjectId, ref: User.name, unique: true, required: true })

    data: FinanceApplicationUser[];
}
