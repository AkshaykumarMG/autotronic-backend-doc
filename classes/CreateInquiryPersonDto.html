import { Types } from "mongoose";
import { Inquiry, InquiryType } from "../schema/inquiry.schema";
import { IsBoolean, IsDateString, IsDefined, IsEmail, IsEnum, IsMongoId, IsNotEmpty, IsNumber, IsOptional, IsString, ValidateIf } from "class-validator";
import { Car } from "../../car/schema/car.schema";
import { User } from "../../../modules/user/schema/user.schema";

/**
 * Data Transfer Object for creating an inquiry from a person.
 * This DTO implements the Inquiry interface and includes validation decorators.
 */
export class CreateInquiryPersonDto implements Inquiry {
    /**
     * The car associated with the inquiry.
     * @type {Car | Types.ObjectId}
     */
    @IsMongoId()
    @IsNotEmpty()
    @IsDefined()
    car: Car | Types.ObjectId;

    /**
     * The user associated with the inquiry.
     * @type {User | Types.ObjectId}
     */
    @IsMongoId()
    @IsNotEmpty()
    @IsDefined()
    user: User | Types.ObjectId;

    /**
     * The type of inquiry.
     * @type {InquiryType}
     */
    @IsEnum(InquiryType)
    type: InquiryType;

    /**
     * The username of the person making the inquiry.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @IsDefined()
    username?: string;

    /**
     * The email address of the person making the inquiry.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @IsDefined()
    @IsEmail()
    email: string;

    /**
     * The phone number of the person making the inquiry.
     * @type {number}
     */
    @IsNumber()
    @IsNotEmpty()
    phone: number;

    /**
     * The message content of the inquiry.
     * @type {string}
     */
    @IsString()
    @IsNotEmpty()
    @IsDefined()
    message: string;

    /**
     * The date for the appointment, if the inquiry type is an appointment.
     * @type {Date}
     */
    @ValidateIf(o => o.type === InquiryType.appointment)
    @IsDateString()
    @IsOptional()
    date: Date;

    /**
     * The postal code of the person making the inquiry.
     * @type {string}
     */
    @IsString()
    @IsOptional()
    postal_code: string;

    /**
     * Indicates whether the inquiry is active or not.
     * @type {boolean}
     */
    @IsBoolean()
    @IsOptional()
    active?: boolean;
}
