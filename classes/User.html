import { Prop, Schema, SchemaFactory } from "@nestjs/mongoose";
import mongoose, { SchemaTypes, Types } from "mongoose";

// hashing password
import * as bcrypt from 'bcrypt';

// relationship
import { Store } from "../../store/schema/store.schema";
import { Car } from "../../car/schema/car.schema";

export interface Address {
    unit: number,
    street: string,
    city: string,
    state: string,
    country: string,
    postal_code: string
}

const AddressSchema = new mongoose.Schema({
    unit: { type: Number, require: true },
    street: { type: String, require: true },
    city: { type: String, require: true },
    state: { type: String, require: true },
    country: { type: String, require: true },
    postal_code: { type: String, require: true },
}, { _id: false });

export interface location {
    type: string,
    coordinates: Array<number>
}

const LocationSchme = new mongoose.Schema({
    type: {
        type: String,
        enum: ['Point'],
        default: 'Point'
    },
    coordinates: {
        type: [Number],
        default: [0, 0]
    }
}, { _id: false });

export enum UserRole {
    Person = 'person',
    Dealer = 'dealer',
    SuperAdmin = 'superadmin',
    Admin = 'admin'
}

/**
 * Represents a user in the system.
 * 
 * @remarks
 * This class is decorated with @Schema to define the MongoDB schema for the User model.
 * It includes various properties that describe a user's attributes and relationships.
 */
@Schema({ timestamps: true, versionKey: false })
export class User {
    /** The unique identifier for the user. */
    _id?: mongoose.Types.ObjectId;
    
    /** 
     * The role of the user in the system.
     * @remarks This is a required field and must be one of the defined UserRole enum values.
     */
    @Prop({ type: String, enum: UserRole, required: true })
    role: UserRole;

    /** The username of the user. */
    @Prop({ type: String })
    username: string;

    /**
     * The email address of the user.
     * @remarks This is a required field and must be unique. It is also indexed for faster queries.
     */
    @Prop({ type: String, required: true, unique: true, index: true })
    email: string;

    /**
     * The password of the user.
     * @remarks This is a required field but is not selected by default in queries for security reasons.
     */
    @Prop({ type: String, required: true, select: false })
    password: string;

    /**
     * The phone number of the user.
     * @remarks This is a required field and must be unique. It is also indexed for faster queries.
     */
    @Prop({ type: Number, unique: true, index: true, required: true })
    phone: number;

    /**
     * Indicates whether the user account is active.
     * @remarks This is a required field.
     */
    @Prop({ type: Boolean, required: true })
    active?: boolean

    /**
     * Additional keys or tokens associated with the user.
     * @remarks This is a required field but is not selected by default in queries for security reasons.
     */
    @Prop({ type: SchemaTypes.Mixed, required: true, select: false })
    keys?: Object;

    /**
     * The stores associated with the user.
     * @remarks This is an array of Store references or ObjectIds.
     */
    @Prop({ type: [{ type: mongoose.Schema.Types.ObjectId, ref: Store.name}] })
    store?: Store[] | Types.ObjectId[];

    /** The address of the user. */
    @Prop({ type: AddressSchema })
    address?: Address;

    /**
     * The geographical location of the user.
     * @remarks This field is indexed as a 2dsphere for geospatial queries.
     */
    @Prop({ type: LocationSchme, index: '2dsphere' })
    location?: location;

    /**
     * The cars favorited by the user.
     * @remarks This is an array of Car references or ObjectIds.
     */
    @Prop({ type: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Car' }] })
    favourite?: Car[] | Types.ObjectId[];
}

export const UserSchema = SchemaFactory.createForClass(User);

/**
 * Pre-save hook for the User schema to hash the user's password before saving.
 * 
 * This function is executed before saving a User document to the database.
 * It checks if a password is present and hashes it using bcrypt with a salt factor of 12.
 * 
 * @param {Function} next - The callback function to be called after password hashing is complete.
 *                          It should be called with an error if one occurs, or with no arguments if successful.
 * 
 * @returns {void} This function doesn't return anything directly, but calls the next() function to continue the save process.
 * 
 * @throws {Error} If there's an error during the password hashing process, it will be passed to the next() function.
 */
UserSchema.pre<User>('save', function (next) {
    const user = this;
    if (user.password) {
        bcrypt.hash(user.password, 12, function (err, hash) {
            if (err) {
                return next(err);
            }
            user.password = hash;
            next();
        });
    }
});
