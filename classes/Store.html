import { Prop, Schema, SchemaFactory } from "@nestjs/mongoose";
import mongoose, { SchemaTypes, Types } from "mongoose";

// relationship
import { Media } from "../../car/schema/car.schema";
import { RemoteService } from "../../remote-service/schema/remote-service.schema";
import { User } from "../../user/schema/user.schema";
import { Award } from "../../award/schema/award.schema";

export interface Address {
    unit: number,
    street: string,
    city: string,
    state: string,
    country: string,
    postal_code: string
}

const AddressSchema = new mongoose.Schema({
    unit: { type: Number, require: true },
    street: { type: String, require: true },
    city: { type: String, require: true },
    state: { type: String, require: true },
    country: { type: String, require: true },
    postal_code: { type: String, require: true },
}, { _id: false });

export interface location {
    // lat: number,
    // lng: number,
    type: string,
    coordinates: Array<number>
}

const LocationSchme = new mongoose.Schema({
    // lat: { type: Number, require: true },
    // lng: { type: Number, require: true },
    type: {
        type: String, // Don't do `{ location: { type: String } }`
        enum: ['Point'], // 'location.type' must be 'Point'
        default: 'Point'
    },
    coordinates: {
        type: [Number],
        default: [0, 0],
        // index: '2d'
    }
}, { _id: false });

export enum TimelineRole {
    dealer = 'dealer',
    agent = 'agent',
    manager = "manager"
}

const TimelineSchme = new mongoose.Schema({
    description: { type: String, require: true },
    note: { type: String, require: true },
    user: { type: mongoose.Schema.Types.ObjectId, ref: 'User', require: true },
    role: { type: String, enum: TimelineRole, require: true },
    duration: { type: Date, default: new Date() },
}, { _id: false });

export interface Timeline {
    description: string,
    role: TimelineRole,
    note?: string,
    user: string
}

export interface Time {
    monday: { in: string, out: string, close: boolean },
    tuesday: { in: string, out: string, close: boolean },
    wednesday: { in: string, out: string, close: boolean },
    thusday: { in: string, out: string, close: boolean },
    friday: { in: string, out: string, close: boolean },
    saturday: { in: string, out: string, close: boolean },
    sunday: { in: string, out: string, close: boolean },
}

const TimeSchme = new mongoose.Schema({
    monday: { type: SchemaTypes.Mixed, require: true },
    tuesday: { type: SchemaTypes.Mixed, require: true },
    wednesday: { type: SchemaTypes.Mixed, require: true },
    thusday: { type: SchemaTypes.Mixed, require: true },
    friday: { type: SchemaTypes.Mixed, require: true },
    saturday: { type: SchemaTypes.Mixed, require: true },
    sunday: { type: SchemaTypes.Mixed, require: true },
}, { _id: false });

export enum StoreStatus {
    assign = 'assign',
    not_assign = 'not_assign',
    in_process = 'in_process',
    verified = 'verified',
    rejected = 'rejected',
    action_required = 'action_required'
}

/**
 * Represents a store in the system.
 * 
 * This class defines the structure and properties of a store, including its
 * address, contact information, services, and operational details.
 */
@Schema({ timestamps: true, versionKey: false })
export class Store {
    /** The unique identifier for the store. */
    _id?: mongoose.Types.ObjectId;

    /** The physical address of the store. */
    @Prop({ type: AddressSchema, required: true })
    address: Address;

    /** A list of timeline events related to the store. */
    @Prop({ type: [TimelineSchme] })
    timeline?: Timeline[];

    /** The user associated with this store, typically the owner or manager. */
    @Prop({ type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true })
    user: User | Types.ObjectId;

    /** The store's contact phone number. */
    @Prop({ type: Number })
    phone: number;

    /** A unique alias or identifier for the store. */
    @Prop({ type: String, required: true, unique: true, index: true })
    alias: string;

    /** The store's website URL. */
    @Prop({ type: String })
    website: string;

    /** The store's contact email address. */
    @Prop({ type: String, require: true })
    email: string;

    /** The store's business license number. */
    @Prop({ type: String, require: true })
    license_number: string;

    /** A link to the store's Google review page. */
    @Prop({ type: String })
    google_review: string;

    /** The official name of the store. */
    @Prop({ type: String })
    name: string;

    /** The geographical location of the store. */
    @Prop({ type: LocationSchme, index: '2dsphere', required: true })
    location: location;

    /** A list of custom services offered by the store. */
    @Prop({ type: Array<String> })
    custom_services: Array<String>;

    /** The store's logo or brand image. */
    @Prop({ type: SchemaTypes.Mixed, required: true })
    logo: Media;

    /** A list of remote services associated with the store. */
    @Prop({ type: [{ type: mongoose.Schema.Types.ObjectId, ref: RemoteService.name }] })
    remote_services: RemoteService[] | Types.ObjectId[];

    /** A list of awards received by the store. */
    @Prop({ type: [{ type: mongoose.Schema.Types.ObjectId, ref: Award.name }] })
    award?: Award[] | Types.ObjectId[];

    /** The store's operational hours. */
    @Prop({ type: TimeSchme, required: true })
    time: Time;

    /** The store's finance Annual Percentage Rate (APR). */
    @Prop({ type: Number })
    finance_apr: number;

    /** The current status of the store. */
    @Prop({ type: String, enum: StoreStatus, index: true, default: StoreStatus.not_assign })
    status?: string;
}

export const StoreSchema = SchemaFactory.createForClass(Store);

StoreSchema.index({ location: "2dsphere" });
