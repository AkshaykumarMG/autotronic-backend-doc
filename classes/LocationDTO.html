/**
 * Represents a data transfer object (DTO) for creating a new car.
 * This class is used to validate and structure the data sent when creating a new car.
 *
 * @implements {Car} - The class implements the Car interface.
 */
export class CreateCarDto implements Car {

  /**
   * The category of the vehicle.
   *
   * @type {VahicleCategory}
   */
  @IsDefined()
  @IsEnum(VahicleCategory)
  category: VahicleCategory;

  /**
   * The make of the vehicle.
   *
   * @type {string}
   */
  @IsString()
  @IsDefined()
  make: string;

  /**
   * The model of the vehicle.
   *
   * @type {string}
   */
  @IsString()
  @IsDefined()
  model: string;

  /**
   * The manufacturing year of the vehicle.
   *
   * @type {number}
   */
  @IsNumber()
  @IsDefined()
  year: number;

  /**
   * The store associated with the vehicle.
   *
   * @type {Store | Types.ObjectId}
   */
  @IsString()
  @IsDefined()
  store: Store | Types.ObjectId;

  /**
   * The Vehicle Identification Number (VIN) of the vehicle.
   *
   * @type {string}
   */
  @IsString()
  @IsDefined()
  vin_number: string;

  /**
   * The stock number of the vehicle, used when VIN is not available.
   *
   * @type {string}
   */
  @ValidateIf(o => !o.vin_number && o.vin_number === null)
  @IsString()
  @IsDefined()
  stock_number: string;

  /**
   * The condition of the vehicle.
   *
   * @type {VahicleCondition}
   */
  @IsDefined()
  @IsEnum(VahicleCondition)
  condition: VahicleCondition;

  /**
   * The type of the seller.
   *
   * @type {UserType}
   */
  @IsDefined()
  @IsEnum(UserType)
  seller_type: UserType;

  /**
   * The current status of the vehicle.
   *
   * @type {VahicleStatus}
   */
  @IsDefined()
  @IsEnum(VahicleStatus)
  status: VahicleStatus;

  /**
   * The vehicle's publication status.
   *
   * @type {VahiclePublish}
   */
  @IsDefined()
  @IsEnum(VahiclePublish)
  publish_to: VahiclePublish;

  /**
   * The asking price of the vehicle.
   *
   * @type {number}
   */
  @IsNumber()
  asking_price: number;

  /**
   * The sales price of the vehicle.
   *
   * @type {number}
   * @optional
   */
  @IsNumber()
  @IsOptional()
  sales_price: number;

  /**
   * The city fuel economy of the vehicle.
   *
   * @type {number}
   * @optional
   */
  @IsNumber()
  @IsOptional()
  city_fuel_economy: number;

  /**
   * The highway fuel economy of the vehicle.
   *
   * @type {number}
   * @optional
   */
  @IsNumber()
  @IsOptional()
  highway_fuel_economy: number;

  /**
   * The manufacturer's suggested retail price (MSRP) of the vehicle.
   *
   * @type {number}
   * @optional
   */
  @IsNumber()
  @IsOptional()
  msrp: number;

  // @IsObject()
  // @Type(() => LocationDTO)
  // location: LocationDTO;

  /**
   * The invoice price of the vehicle.
   *
   * @type {number}
   * @optional
   */
  @IsNumber()
  @IsOptional()
  invoice_price: number;

  /**
   * The retail price of the vehicle.
   *
   * @type {number}
   * @optional
   */
  @IsNumber()
  @IsOptional()
  retail_price: number;

  /**
   * The wholesale price of the vehicle.
   *
   * @type {number}
   * @optional
   */
  @IsNumber()
  @IsOptional()
  wholesale_price: number;

  /**
   * A detailed description of the vehicle.
   *
   * @type {string}
   */
  @IsString()
  @IsDefined()
  description: string;

  /**
   * The URL or identifier for the vehicle's Carfax report.
   *
   * @type {string}
   * @optional
   */
  @IsString()
  @IsOptional()
  carafax: string;

  /**
   * A detailed description of the vehicle's warranty.
   *
   * @type {string}
   */
  @IsString()
  @IsDefined()
  warrenty_description: string;

  /**
   * The current odometer reading of the vehicle.
   *
   * @type {number}
   */
  @IsNumber()
  odometer: number;

  /**
   * The date when the vehicle is expected to be in stock.
   *
   * @type {Date}
   * @optional
   */
  @IsOptional()
  @IsDateString()
  in_stock_date: Date;

  /**
   * An array of images associated with the vehicle.
   *
   * @type {Media[]}
   * @optional
   * @minimum 1
   */
  @IsOptional()
  @IsDefined()
  @IsArray()
  @ArrayMinSize(1)
  @ValidateNested({ each: true })
  @Type(() => MediaDTO)
  images: Media[];

  /**
   * An array of videos associated with the vehicle.
   *
   * @type {Media[]}
   * @optional
   */
  @IsArray()
  @IsOptional()
  @ValidateNested({ each: true })
  @Type(() => MediaDTO)
  videos: Media[];

  /**
   * The URL for a video associated with the vehicle.
   *
   * @type {string}
   * @optional
   */
  @IsString()
  @IsOptional()
  video_url: string;

  /**
   * Additional window stickers for the vehicle.
   *
   * @type {Object}
   * @optional
   */
  @IsObject()
  @IsOptional()
  window_stickers: Object;
}
