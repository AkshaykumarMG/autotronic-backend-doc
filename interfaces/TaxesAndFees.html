import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import mongoose, { HydratedDocument, SchemaTypes, Types } from 'mongoose';
import { User } from '../../user/schema/user.schema';
import { Store } from '../../store/schema/store.schema';

export type CarDocument = HydratedDocument<Car>;

export enum VahicleCategory {
  car = 'car',
  truck = 'truck',
  bus = 'bus',
  checkbox = 'checkbox',
  'commercial_vehicals' = 'commercial_vehicals'
}

export enum VahicleCondition {
  new = 'new',
  used = 'used',
  damaged = 'damaged'
}

export enum VahicleStatus {
  live = 'live',
  removed = 'removed',
  sold = 'sold',
  // pending = 'pending'
}

export enum VahiclePublish {
  'euro_exoti_car' = 'euro_exoti_car',
  none = 'none'
}

export enum MediaField {
  'image' = 'images',
  'video' = 'videos'
}

export enum UserType {
  'dealer' = 'dealer',
  'private_person' = 'private_person'
}

export interface Media {
  url: string,
  name: string,
  path: string,
  field: MediaField,
  position: number,
  date: Date
}

export interface TaxesAndFees {
  type: string,
  value: number,
  mode: string
}

export interface location {
  type: string,
  coordinates: Array<number>
}

const LocationSchme = new mongoose.Schema({
  type: {
    type: String, // Don't do `{ location: { type: String } }`
    enum: ['Point'], // 'location.type' must be 'Point'
    default: 'Point'
  },
  coordinates: {
    type: [Number],
    default: [0, 0],
    // index: '2d'
  }
}, { _id: false });

@Schema({ timestamps: true, versionKey: false })
/**
 * Mongoose schema for a Car document.
 * This schema represents a car's details and attributes.
 */
@Schema({ timestamps: true, versionKey: false })
export class Car {

  // Unique identifier for the car document.
  _id?: mongoose.Types.ObjectId;

  // Name of the car.
  @Prop({ type: String, required: true })
  name?: string;

  // User who created the car document.
  @Prop({ type: mongoose.Schema.Types.ObjectId, ref: User.name })
  user?: User | Types.ObjectId;

  // Store associated with the car.
  @Prop({ type: mongoose.Schema.Types.ObjectId, ref: Store.name })
  store: Store | Types.ObjectId;

  // Category of the car (e.g., car, truck, bus).
  @Prop({ type: String, required: [true, 'Category is required'], index: true, enum: VahicleCategory })
  category: VahicleCategory;

  // Make (company) of the car.
  @Prop({ type: String, required: [true, 'Make(Company) is required'], index: true })
  make: string;

  // Model of the car.
  @Prop({ type: String, required: [true, 'Model is required'], index: true })
  model: string;

  // Year of manufacture of the car.
  @Prop({ type: Number, required: [true, 'Condition is required'], index: true })
  year: number;

  // VIN (Vehicle Identification Number) of the car.
  @Prop({ type: String, set: (value: string) => value.toUpperCase() })
  vin_number: string;

  // Stock number of the car.
  @Prop({
    type: String,
    required: function () { return !this.vin_number },
    // set: (value: string) => value.toUpperCase()
  })
  stock_number: string;

  // Condition of the car (e.g., new, used, damaged).
  @Prop({ type: String, required: [true, 'Condition is required'], index: true, enum: VahicleCondition })
  condition: VahicleCondition;

  // Status of the car (e.g., live, removed, sold).
  @Prop({ type: String, required: [true, 'Status is required'], enum: VahicleStatus })
  status: VahicleStatus;

  // Type of seller (e.g., dealer, private person).
  @Prop({ type: String, required: true, enum: UserType })
  seller_type: UserType;

  // Publish options for the car (e.g., euro_exoti_car, none).
  @Prop({ type: String, enum: VahiclePublish, default: VahiclePublish.euro_exoti_car })
  publish_to: VahiclePublish;

  // Asking price of the car.
  @Prop({ type: Number, required: [true, 'Asking Price is required'] })
  asking_price: number;

  // Sales price of the car.
  @Prop({ type: Number })
  sales_price: number;

  // MSRP (Manufacturer's Suggested Retail Price) of the car.
  @Prop({ type: Number })
  msrp: number;

  // Invoice price of the car.
  @Prop({ type: Number })
  invoice_price: number;

  // Retail price of the car.
  @Prop({ type: Number })
  retail_price: number;

  // Wholesale price of the car.
  @Prop({ type: Number })
  wholesale_price: number;

  // Description of the car.
  @Prop({ type: String, required: [true, 'Description is required'] })
  description: string;

  // Warranty description of the car.
  @Prop({ type: String, required: [true, 'Warrenty Description is required'] })
  warrenty_description: string;

  // Odometer reading of the car.
  @Prop({ type: Number, required: [true, 'Odometer is required'] })
  odometer: number;

  // City fuel economy of the car.
  @Prop({ type: Number })
  city_fuel_economy: number;

  // Highway fuel economy of the car.
  @Prop({ type: Number }) 
  highway_fuel_economy: number;

  // In-stock date of the car.
  @Prop({ type: Date, required: [true, 'In Stock Date is required'] })
  in_stock_date: Date;

  // Images of the car.
  @Prop({ type: Array<Media>, default: [] })
  images?: Media[] = []

  // Videos of the car.
  @Prop({ type: Array<Media>, default: [] })
  videos?: Media[] = []

  // Location of the car.
  @Prop({ type: LocationSchme, index: '2dsphere' })
  location?: location;

  // Video URL of the car.
  @Prop({ type: String })
  video_url: string;

  // Carfax URL of the car.
  @Prop({ type: String })
  carafax: string;

  // Window stickers of the car.
  @Prop({ type: SchemaTypes.Mixed, of: String })
  window_stickers: Object;

  // Active status of the car.
  @Prop({ type: Boolean, default: true, index: true })
  active?: boolean;

  // User who created the car document.
  @Prop({ type: mongoose.Schema.Types.ObjectId, ref: User.name, required: true })
  createdBy?: User | Types.ObjectId;

  // Number of views the car has received.
  @Prop({ type: Number, default: 0 })
  user_views?: number

  // Taxes and fees associated with the car.
  @Prop({ type: Array<TaxesAndFees>, required: true })
  taxes_fees?: Array<TaxesAndFees>;

  // Users who have favorited the car.
  @Prop({ type: [{ type: mongoose.Schema.Types.ObjectId, ref: 'User' }] })
  favourite?: User[] | Types.ObjectId[];

  // Price history of the car.
  @Prop({ type: Array<Number> })
  price_history?: Array<number>;
}


export const CarSchema = SchemaFactory.createForClass(Car);

CarSchema.index({ location: "2dsphere" });
