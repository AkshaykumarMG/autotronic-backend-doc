import { Injectable } from '@nestjs/common';
import { CreateNotificationDto } from './dto/create-notification.dto';
// import { UpdateNotificationDto } from './dto/update-notification.dto';
import { UpdateReadNotificationDto } from './dto/update-read-notification.dto';

// mongoose
import { Notification } from './schema/notification.schema';
import { Model, Types } from 'mongoose';
import { InjectModel } from '@nestjs/mongoose';

@Injectable()
export class NotificationService {

  constructor(
    @InjectModel(Notification.name) private notificationSchema: Model<Notification>
  ) {}

  /**
   * The function creates a new notification using the provided data and saves it to the database
   * asynchronously.
   * @param {CreateNotificationDto} createNotificationDto - `createNotificationDto` is an object that
   * contains the data needed to create a new notification. It likely includes properties such as the
   * notification message, recipient, timestamp, and any other relevant information required to create
   * a notification.
   * @returns The `create` function is returning a promise that resolves to the saved notification
   * object created from the `createNotificationDto` input.
   */
  async create(createNotificationDto: CreateNotificationDto) {
    const createdNotification = new this.notificationSchema(createNotificationDto);
    return await createdNotification.save();
  }

  /**
   * This function asynchronously retrieves a specified number of notifications sorted by creation
   * date, with pagination support.
   * @param {number} [page=0] - The `page` parameter is used to specify the page number of results to
   * retrieve. It is set to a default value of 0, meaning that by default, the function will retrieve
   * results starting from the first page.
   * @param {number} [limit=10] - The `limit` parameter specifies the maximum number of documents that
   * the `findAll` method will return in a single query. In this case, the default value for `limit` is
   * set to 10, meaning that by default, the method will return up to 10 documents per query unless a
   * different
   * @returns An array of notifications sorted by createdAt date in descending order, limited by the
   * specified limit and paginated based on the page number provided.
   */
  async findAll(id: Types.ObjectId, page: number = 0, limit: number = 5) {
    
    const unread = await this.notificationSchema.countDocuments({ read: false, to: id });
    const total = await this.notificationSchema.countDocuments({ to: id });
    const data = await this.notificationSchema.find({ to: id }).sort({ createdAt: -1 }).limit(limit).skip(page * limit).exec();

    // get id's from fetched record
    // const notIDs = data.map((notification: Notification) => {  return notification._id });
    // set flag notification read to unread
    // await this.notificationSchema.updateMany({ _id: { $in: notIDs }}, { $set: { read: true }}, { multi: true });

    // return data and count
    return { data, total, unread };
  }

  /**
   * This TypeScript function updates the read status of a notification document in a MongoDB database.
   * @param id - The `id` parameter is of type `Types.ObjectId`, which is typically used to represent a
   * unique identifier for a document in a MongoDB database. It is commonly used as the primary key for
   * documents in a collection.
   * @param {UpdateReadNotificationDto} updateReadNotificationDto - The `updateReadNotificationDto`
   * parameter is an object that contains the information needed to update the read status of a
   * notification. It likely has a property named `read` that specifies the new read status value for
   * the notification.
   * @returns The `updateReadSatus` function is returning a promise that resolves to the result of
   * updating the notification document in the database with the specified `_id` using the
   * `notificationSchema`. The document's `read` field is updated with the value provided in the
   * `updateReadNotificationDto.read` parameter.
   */
  async updateReadSatus(id: Types.ObjectId, updateReadNotificationDto: UpdateReadNotificationDto) {
    return await this.notificationSchema.findByIdAndUpdate({ _id: id }, { $set: { read: updateReadNotificationDto.read }});
  }
}
