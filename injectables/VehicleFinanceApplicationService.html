// user add on new user registration

export class VehicleFinanceApplicationService {

  constructor(
    private readonly mailService: MailService,
    private readonly userService: UserService,
    private readonly httpService: HttpService,
    private readonly authenticationService: AuthenticationService,
    private readonly financeApplicationService: FinanceApplicationService,
    @InjectModel(VehicleFinanceApplication.name) private readonly vehicleFinanceApplicationSchema: Model<VehicleFinanceApplication>
  ) { }

  /**
   * Creates a new Vehicle Finance Application.
   *
   * This function handles the creation of a new vehicle finance application, including checking for existing finance applications,
   * updating credit application data, and saving user preferences. It also generates a random result for the application.
   *
   * @param createVehicleFinanceApplicationDto - The data required to create a new vehicle finance application.
   * This object should contain the following properties:
   * - user: The unique identifier of the user associated with the application.
   * - monthly_payment: The monthly payment for the vehicle finance application.
   * - annual_income: The annual income of the primary applicant.
   * - estimated_credit_score: The estimated credit score of the primary applicant.
   * - save_preferance: A boolean indicating whether to save the provided credit application data.
   *
   * @throws Will throw an error if the user already has a finance application without provided credit application data,
   * or if there is a database connection issue.
   *
   * @returns A Promise that resolves to an object containing the following properties:
   * - _id: The unique identifier of the newly created vehicle finance application.
   * - result: The result of the vehicle finance application, generated using the `generateRandomResult` method.
   * - estimated_credit_score: The estimated credit score of the primary applicant.
   * - annual_income: The annual income of the primary applicant.
   * - monthly_payment: The monthly payment for the vehicle finance application.
   */
  async create(createVehicleFinanceApplicationDto: CreateVehicleFinanceApplicationDto) {

    // check we have payment monthly, annual income, estimated credit score in requested vehicle finance application
    const isNewApplication = !createVehicleFinanceApplicationDto.monthly_payment && !createVehicleFinanceApplicationDto.annual_income && !createVehicleFinanceApplicationDto.estimated_credit_score;;

    // check if user already have finance application
    const financeApplication = await this.financeApplicationService.findOneByUserId(createVehicleFinanceApplicationDto.user as Types.ObjectId);
    if (!financeApplication && isNewApplication) {
      throw new HttpException('Missing Information', HttpStatus.NOT_FOUND);
    }

    // if save preferance then update credit application
    if (createVehicleFinanceApplicationDto.save_preferance) {
      // if user already have finance application then update otherwise create new finance application
      if (financeApplication) {
        const commonFields = {
          estimated_credit_score: createVehicleFinanceApplicationDto.estimated_credit_score,
          gross_annual_income: createVehicleFinanceApplicationDto.annual_income,
          monthly_payment: createVehicleFinanceApplicationDto.monthly_payment
        }
        await this.financeApplicationService.updatePrimaryApplicantCommonFields(financeApplication._id, commonFields);
      } else {
        await this.financeApplicationService.create({
          user: createVehicleFinanceApplicationDto.user,
          data: [{
            monthly_payment: createVehicleFinanceApplicationDto.monthly_payment,
            gross_annual_income: createVehicleFinanceApplicationDto.annual_income,
            estimated_credit_score: createVehicleFinanceApplicationDto.estimated_credit_score,
          }]
        } as any);
      }
    }

    if (financeApplication) {
      // update  necessary field values
      createVehicleFinanceApplicationDto.estimated_credit_score = financeApplication.data[0].estimated_credit_score;
      createVehicleFinanceApplicationDto.annual_income = financeApplication.data[0].gross_annual_income;
      createVehicleFinanceApplicationDto.monthly_payment = financeApplication.data[0].monthly_payment;
    }


    // creatre result request body
    const body = {
      "age": 31,
      "annual_income": createVehicleFinanceApplicationDto.annual_income,
      "house_status": true,
      "car_loan_status": true,
      "bankrupt": false,
      "work_type": "full_time",
      "residency_status": "canadian_citizen",
      "down_payment": createVehicleFinanceApplicationDto.down_payment,
      "estimated_credit_score": createVehicleFinanceApplicationDto.estimated_credit_score,
      "monthly_payment": createVehicleFinanceApplicationDto.monthly_payment,
      "year": createVehicleFinanceApplicationDto.year,
      "price": createVehicleFinanceApplicationDto.price,
    };

    // console.log('vehicle finance application created', createVehicleFinanceApplicationDto, body);

    // call external api for application result
    await lastValueFrom(this.httpService.post(`${config.HEROKU_PYTHON_URL}finance`, body))
      .then((response) => {
        // console.log('result from api', response.data);
        // save application result in Vehicle Finance Application
        createVehicleFinanceApplicationDto.result = response.data;
      }).catch((error) => {
        throw new HttpException('Kindly Try Again Later.', HttpStatus.BAD_REQUEST);
      });

    // generate result of the application
    // const appResult = this.generateRandomResult();

    // save vehicle finance application in db
    const createdVehicleFinanceApplication = new this.vehicleFinanceApplicationSchema(createVehicleFinanceApplicationDto);
    const result = await createdVehicleFinanceApplication.save();

    return {
      _id: result._id,
      result: createVehicleFinanceApplicationDto.result,
      estimated_credit_score: createVehicleFinanceApplicationDto.estimated_credit_score,
      annual_income: createVehicleFinanceApplicationDto.annual_income,
      monthly_payment: createVehicleFinanceApplicationDto.monthly_payment
    };
  }

  /** 
   * Creates a new sub-vehicle finance application associated with the given parent application ID.
   *
   * @param id - The unique identifier of the parent Vehicle Finance Application.
   * @param createSubVehicleFinanceApplicationDto - The data required to create a new sub-vehicle finance application.
   *
   * @throws Will throw an error if the parent Vehicle Finance Application is not found.
   *
   * @returns A Promise that resolves to void.

  async createSubVehicleFinanceApplication(id: Types.ObjectId, createSubVehicleFinanceApplicationDto: CreateSubVehicleFinanceApplicationDto) {
    const vehicleFinanceApplication = await this.vehicleFinanceApplicationSchema.findById(id);
    if (!vehicleFinanceApplication) {
      throw new HttpException('Vehicle Finance Application not found', HttpStatus.NOT_FOUND);
    }

    // add modified search with results
    vehicleFinanceApplication.data.push(createSubVehicleFinanceApplicationDto);
    await vehicleFinanceApplication.save();
  }

  /**
   * Creates a direct result for a finance application by calling an external API.
   * 
   * This function takes the provided direct result data, constructs a request body,
   * sends it to an external finance API, and returns the maximum values for amount,
   * APR, and term from the API response.
   *
   * @param createDirectResultDto - The DTO containing the data for creating a direct result.
   * @param createDirectResultDto.annual_income - The annual income of the applicant.
   * @param createDirectResultDto.estimated_credit_score - The estimated credit score of the applicant.
   * @param createDirectResultDto.monthly_payment - The proposed monthly payment.
   * 
   * @returns A Promise that resolves to an object containing the maximum values for:
   *          - amount: The maximum loan amount
   *          - apr: The maximum Annual Percentage Rate
   *          - term: The maximum loan term
   * 
   * @throws {HttpException} With a 'BAD_REQUEST' status if the external API call fails.
   */
  async createDirectResult(createDirectResultDto: CreateDirectResultDto) {
    try {
      const body = {
        "age": 31,
        "annual_income": createDirectResultDto.annual_income,
        "down_payment": 0,
        "estimated_credit_score": createDirectResultDto.estimated_credit_score,
        "monthly_payment": createDirectResultDto.monthly_payment,
        "year": new Date().getFullYear(),
        "price": 50000,
      }
      // call external api for application result
      const result = await lastValueFrom(this.httpService.post(`${config.HEROKU_PYTHON_URL}finance`, body));
      return {
        amount:Math.max(...result.data.map(d => d.amount)),
        apr: Math.max(...result.data.map(d => d.apr)),
        term: Math.max(...result.data.map(d => d.term))
      };
    } catch (error) {
      throw new HttpException('Kindly Try Again Later.', HttpStatus.BAD_REQUEST);
    }
  }

  /**
 * Handles the creation of a new user and their associated vehicle finance application.
 * This function performs several tasks, including user registration, vehicle finance application creation,
 * JWT token generation, authentication data storage, and sending a welcome email.
 *
 * @param request - The incoming HTTP request object, containing headers like IP and user-agent.
 * @param createUserVehicleFinanceApplicationDto - The data required to create a new user and vehicle finance application.
 *
 * @returns A Promise that resolves to an object containing the following properties:
 * - success: A boolean indicating whether the operation was successful.
 * - token: The JWT token generated for the newly created user.
 * - auth: The unique identifier of the authentication data stored in the database.
 * - store: The store associated with the user.
 * - vehicleAppID: The unique identifier of the newly created vehicle finance application.
 * - data: An object containing the user's identifier, email, username, and role type.
 * - result: An object containing the result of the vehicle finance application, including score, status, term, amount, and apr.
 *
 * @throws Will throw an error if the email or phone already exist in the database, or if there is a database connection issue.
 */
  async createWithUserRegistration(request: Request, createUserVehicleFinanceApplicationDto: CreateUserVehicleFinanceApplicationDto) {

    // check user is exist with email and phone
    const isExistEmail = await this.userService.findOneByEMail(createUserVehicleFinanceApplicationDto.email);

    if (isExistEmail) {
      throw new HttpException('Email already exist, try login!', HttpStatus.FOUND);
    }

    // check phone is exist
    const isExistPhone = await this.userService.findOneByPhone(createUserVehicleFinanceApplicationDto.phone);
    if (isExistPhone) {
      throw new HttpException('Phone already exist, try login!', HttpStatus.FOUND);
    }

    // generate auto password
    const password = Math.random().toString(36).slice(-8);

    // create user account
    const userData: CreateUserDto = {
      username: createUserVehicleFinanceApplicationDto.username,
      email: createUserVehicleFinanceApplicationDto.email,
      password,
      phone: createUserVehicleFinanceApplicationDto.phone,
      active: true,
      role: UserRole.Person
    }

    // add user in db
    const user = await this.userService.create(userData);

    // create vehicle finance application
    const vehicleFinanceApplicationData = { ...createUserVehicleFinanceApplicationDto, user: user._id };

    // store new user finance application default
    vehicleFinanceApplicationData.save_preferance = true;

    // save in db vehicle info
    const createdVehicleFinanceApplication = await this.create(vehicleFinanceApplicationData);

    // find user with credential keys
    const userExist = await this.userService.findOneForVerify(user._id);

    // store data in token
    const data = {
      identifier: userExist._id,
      email: userExist.email,
      username: userExist.username,
    };

    //generate token 
    const token = await this.userService.createJWTToken(userExist.keys, data);

    // get ip 
    const ip = request.headers['x-forwarded-for'] || request.socket.remoteAddress || null;

    // get user-agent
    const user_agent = request.headers['user-agent'];

    // add authentication
    const authentication = new Authentication();
    authentication.email = user.email.trim();
    authentication.token = token;
    authentication.type = AuthType.signin;
    authentication.ip = ip as string;
    authentication.user_agent = user_agent;

    // save auth data in db
    const authResult = await this.authenticationService.createAuthentication(authentication);

    // send mail with credential
    const mailClient = {
      to: user.email,
      from: 'akshay@mechatronic-group.com',
      templateId: 'd-b29caf39344941f8a251fc80a6faeed3',
      dynamicTemplateData: {
        email: user.email,
        password: password,
      },
    };
    await this.mailService.send(mailClient);

    return {
      success: true,
      token: await this.userService.createJWTToken(userExist.keys, data),
      auth: String(authResult._id),
      store: userExist.store,
      vehicleAppID: createdVehicleFinanceApplication._id,
      data: {
        identifier: userExist._id,
        email: userExist.email,
        username: userExist.username,
        type: this.userService.getRoleCodeByRoleName(userExist.role)
      },
      result: {
        data: createdVehicleFinanceApplication.result,
        down_payment: vehicleFinanceApplicationData.down_payment,
        estimated_credit_score: vehicleFinanceApplicationData.estimated_credit_score,
        annual_income: vehicleFinanceApplicationData.annual_income,
        monthly_payment: vehicleFinanceApplicationData.monthly_payment
      }
    };
  }

  /**
 * Updates a Vehicle Finance Application by its ID.
 *
 * This function handles updating a vehicle finance application, including updating credit application data
 * and generating a random result for the application.
 *
 * @param id - The unique identifier of the Vehicle Finance Application to update.
 * @param updateOtherFieldVehicleDto - The data required to update the Vehicle Finance Application.
 * This object should contain the following properties:
 * - user: The unique identifier of the user associated with the application.
 * - monthly_payment: The monthly payment for the vehicle finance application.
 * - annual_income: The annual income of the primary applicant.
 * - estimated_credit_score: The estimated credit score of the primary applicant.
 * - save_preferance: A boolean indicating whether to save the provided credit application data.
 *
 * @returns A Promise that resolves to an object containing the following properties:
 * - result: The result of the vehicle finance application, generated using the `generateRandomResult` method.
 *
 * @throws Will throw an error if there is a database connection issue or if the ID is not valid.
  */
  async update(id: Types.ObjectId, updateOtherFieldVehicleDto: UpdateOtherFieldVehicleDto) {

    // update primary applicant common fields in finance application table
    if (updateOtherFieldVehicleDto.save_preferance) {
      const financeApplication = await this.financeApplicationService.findOneByUserId(updateOtherFieldVehicleDto.user);
      if (financeApplication) {
        const commonFields = {
          estimated_credit_score: updateOtherFieldVehicleDto.estimated_credit_score,
          gross_annual_income: updateOtherFieldVehicleDto.annual_income,
          monthly_payment: updateOtherFieldVehicleDto.monthly_payment
        }
        await this.financeApplicationService.updatePrimaryApplicantCommonFields(financeApplication._id, commonFields);
      } else {
        await this.financeApplicationService.create({
          user: updateOtherFieldVehicleDto.user,
          data: [{
            estimated_credit_score: updateOtherFieldVehicleDto.estimated_credit_score,
            monthly_payment: updateOtherFieldVehicleDto.monthly_payment,
            gross_annual_income: updateOtherFieldVehicleDto.annual_income,
          }]
        } as any);
      }
    }

    // update vehicle finance application in db with new result
    const result = await this.vehicleFinanceApplicationSchema.findByIdAndUpdate({ _id: id }, { $set: updateOtherFieldVehicleDto });

    // generate result of the application
    // creatre result request body
    const body = {
      "age": updateOtherFieldVehicleDto.age,
      "annual_income": updateOtherFieldVehicleDto.annual_income,
      "house_status": updateOtherFieldVehicleDto.house_status,
      "car_loan_status": updateOtherFieldVehicleDto.car_loan_status,
      "bankrupt": updateOtherFieldVehicleDto.bankrupt,
      "work_type": updateOtherFieldVehicleDto.work_type,
      "residency_status": updateOtherFieldVehicleDto.residency_status,
      "down_payment": updateOtherFieldVehicleDto.down_payment,
      "estimated_credit_score": updateOtherFieldVehicleDto.estimated_credit_score,
      "monthly_payment": updateOtherFieldVehicleDto.monthly_payment,
      "year": result.year,
      "price": result.price,
    };

    // console.log('body', body);

    // call external api for application result
    await lastValueFrom(this.httpService.post(`${config.HEROKU_PYTHON_URL}finance`, body))
      .then((response) => {
        //  console.log('result from api', response.data);
        // save application result in Vehicle Finance Application
        updateOtherFieldVehicleDto.result = response.data;
      }).catch((error) => {
        throw new HttpException('Kindly Try Again Later.', HttpStatus.BAD_REQUEST);
      });

    return { result: updateOtherFieldVehicleDto.result };
  }
