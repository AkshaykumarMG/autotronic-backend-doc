import { HttpException, HttpStatus, Injectable } from '@nestjs/common';
import { CreateAwardDto } from './dto/create-award.dto';
import { UpdateAwardDto } from './dto/update-award.dto';
import { ManageUserAwardDto } from './dto/manage-user-award.dto';

// mongoose
import { Model, Types } from 'mongoose';
import { Award } from './schema/award.schema';
import { InjectModel } from '@nestjs/mongoose';

// aws service to remove file during remove
import { AwsS3Service } from '../../utilities/services/aws_s3/aws_s3.service';

// relationship
import { StoreService } from '../store/store.service';

@Injectable()
export class AwardService {

  constructor(
    private readonly awsS3Service: AwsS3Service,
    private readonly storeService: StoreService,
    @InjectModel(Award.name) private readonly awardSchema: Model<Award>
  ) { }

  /**
   * Creates a new award in the database.
   * 
   * @param createAwardDto - The data transfer object containing the award details to be created.
   * @returns A Promise that resolves to the newly created award document.
   */
  async create(createAwardDto: CreateAwardDto) {
    const createdAward = new this.awardSchema(createAwardDto);
    return await createdAward.save();
  }

  /**
   * Retrieves a paginated list of all awards from the database.
   * 
   * @param page - The page number for pagination. Defaults to 0 if not provided.
   * @param limit - The maximum number of awards to return per page. Defaults to 10 if not provided.
   * @returns A Promise that resolves to an array of Award documents, sorted by creation date in descending order.
   */
  async findAll(page: number = 0, limit: number = 10) {
    return await this.awardSchema.find().sort({ createdAt: -1 }).limit(limit).skip(page * limit).exec();
  }

  /**
   * Retrieves a list of all awards from the database, selecting only the name and ID fields.
   * This function is optimized for populating select dropdowns or similar UI elements.
   *
   * @param page - The page number for pagination. Defaults to 0 if not provided.
   * @param limit - The maximum number of awards to return. Defaults to 10 if not provided.
   * @returns A Promise that resolves to an array of Award documents, containing only the name and ID fields,
   *          sorted by creation date in descending order.
   */
  async findAllForSelect(page: number = 0, limit: number = 10) {
    return await this.awardSchema.find().sort({ createdAt: -1 }).select({ 'name': 1, '_id': 1 }).exec();
  }

  /**
   * Retrieves a single award from the database based on its ID.
   *
   * @param id - The ObjectId of the award to retrieve.
   * @returns A Promise that resolves to the Award document matching the provided ID,
   *          or null if no award is found.
   */
  async findOne(id: Types.ObjectId) {
    return await this.awardSchema.findById(id);
  }

  /**
   * Updates an existing award in the database.
   * If a new image is provided, it deletes the previous image from AWS S3 storage.
   *
   * @param id - The ObjectId of the award to update.
   * @param updateAwardDto - The data transfer object containing the updated award details.
   * @returns A Promise that resolves to the updated Award document.
   */
  async update(id: Types.ObjectId, updateAwardDto: UpdateAwardDto) {
    if (updateAwardDto.image) {
      const previousImage = await this.awardSchema.findById(id);
      await this.awsS3Service.delete(previousImage.image.path + previousImage.image.name)
    }
    return await this.awardSchema.findByIdAndUpdate({ _id: id }, { $set: updateAwardDto });
  }

  /**
   * Adds an award to a store and updates the corresponding store's award list.
   * 
   * This function performs two operations:
   * 1. It adds the store to the award's list of stores.
   * 2. It updates the store's list of awards to include the new award.
   *
   * @param manageUserAwardDto - The data transfer object containing the award and store information.
   * @param manageUserAwardDto.award - The ID of the award to be added to the store.
   * @param manageUserAwardDto.store - The ID of the store to which the award will be added.
   * 
   * @returns A Promise that resolves when both operations are complete.
   * @throws {Error} If there's an issue updating the award or store documents.
   */
  async updateAddAwardToStore(manageUserAwardDto: ManageUserAwardDto) {
    await this.awardSchema.findByIdAndUpdate({ _id: manageUserAwardDto.award }, { $addToSet: { store: manageUserAwardDto.store } });
    await this.storeService.updateAddAndRemoveAward(manageUserAwardDto)
  }

  /**
   * Removes an award from a store and updates the corresponding store's award list.
   * 
   * This function performs two operations:
   * 1. It removes the store from the award's list of stores.
   * 2. It updates the store's list of awards to remove the specified award.
   *
   * @param manageUserAwardDto - The data transfer object containing the award and store information.
   * @param manageUserAwardDto.award - The ID of the award to be removed from the store.
   * @param manageUserAwardDto.store - The ID of the store from which the award will be removed.
   * 
   * @returns A Promise that resolves when both operations are complete.
   * @throws {Error} If there's an issue updating the award or store documents.
   */
  async updateRmoveAwardToStore(manageUserAwardDto: ManageUserAwardDto) {
    await this.awardSchema.findByIdAndUpdate({ _id: manageUserAwardDto.award }, { $pull: { store: manageUserAwardDto.store } });
    await this.storeService.updateAddAndRemoveAward(manageUserAwardDto, 'pull');
  }

  /**
   * Removes an award from the database and deletes its associated image from AWS S3 storage.
   *
   * This function performs two main operations:
   * 1. It retrieves the award's image information and deletes the image file from AWS S3.
   * 2. It removes the award document from the database.
   *
   * @param id - The ObjectId of the award to be removed.
   * @returns A Promise that resolves to the deleted Award document, or null if no award was found.
   * @throws {Error} If there's an issue deleting the image from S3 or removing the award from the database.
   */
  async remove(id: Types.ObjectId) {
    // remove image from bucket
    const previousImage = await this.awardSchema.findById(id);
    await this.awsS3Service.delete(previousImage.image.path + previousImage.image.name);
  
    return await this.awardSchema.findByIdAndDelete(id);
  }
}
