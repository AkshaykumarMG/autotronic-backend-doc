import { Injectable } from '@nestjs/common';
import { CreateRemoteServiceDto } from './dto/create-remote-service.dto';
import { UpdateRemoteServiceDto } from './dto/update-remote-service.dto';
import { RemoteService } from './schema/remote-service.schema';

// mongoose
import { InjectModel } from '@nestjs/mongoose';
import { Model, Types } from 'mongoose';

// service images s3 baucket
import { AwsS3Service } from '../../utilities/services/aws_s3/aws_s3.service';

@Injectable()
export class RemoteServiceService {

  constructor(
    private readonly awsS3Service: AwsS3Service,
    @InjectModel(RemoteService.name) private readonly remoteServiceSchema: Model<RemoteService>
  ) {}

  /**
   * Creates a new remote service entry in the database.
   * 
   * @param createRemoteServiceDto - The data transfer object containing the details of the remote service to be created.
   * @returns A promise that resolves to the newly created remote service document.
   */
  async create(createRemoteServiceDto: CreateRemoteServiceDto) {
    const createdRemoteService = new this.remoteServiceSchema(createRemoteServiceDto);
    return await createdRemoteService.save();
  }

  /**
   * Retrieves a paginated list of remote services from the database.
   * 
   * @param page - The page number to retrieve. Defaults to 0 if not provided.
   * @param limit - The maximum number of items to return per page. Defaults to 10 if not provided.
   * @returns A promise that resolves to an array of remote service documents, sorted by creation date in descending order.
   */
  async findAll(page: number = 0, limit: number = 10) {
    return await this.remoteServiceSchema.find().sort({ createdAt: -1 }).limit(limit).skip(page * limit).exec();
  }

  /**
   * Retrieves all remote services for selection purposes, excluding certain fields.
   * 
   * This method fetches all remote service documents from the database, sorts them
   * by creation date in descending order, and excludes specific fields to provide
   * a streamlined dataset suitable for selection interfaces or dropdowns.
   * 
   * @returns A promise that resolves to an array of remote service documents,
   *          sorted by creation date in descending order, with the 'createdAt',
   *          'updatedAt', 'createdBy', and 'active' fields excluded.
   */
  async findAllForSelection() {
    return await this.remoteServiceSchema.find().sort({ createdAt: -1 }).select(['-createdAt', '-updatedAt', '-createdBy', '-active']).exec();
  }

  /**
   * Retrieves a single remote service document from the database by its ID.
   * 
   * @param id - The unique identifier of the remote service to retrieve.
   *             It should be a valid MongoDB ObjectId.
   * @returns A promise that resolves to the found remote service document,
   *          or null if no document matches the given ID.
   */
  async findOne(id: Types.ObjectId) {
    return await this.remoteServiceSchema.findById(id);
  }

  /**
   * Updates an existing remote service document in the database.
   * If a new image is provided, it deletes the previous image from S3 storage.
   *
   * @param id - The unique identifier of the remote service to update.
   *             It should be a valid MongoDB ObjectId.
   * @param updateRemoteServiceDto - The data transfer object containing the updated details of the remote service.
   * @returns A promise that resolves to the updated remote service document.
   */
  async update(id: Types.ObjectId, updateRemoteServiceDto: UpdateRemoteServiceDto) {
    if(updateRemoteServiceDto.image) {
      const previousImage = await this.remoteServiceSchema.findById(id);
      await this.awsS3Service.delete(previousImage.image.path + previousImage.image.name)
    }
    return await this.remoteServiceSchema.findByIdAndUpdate({ _id: id }, { $set: updateRemoteServiceDto });
  }

  /**
   * Removes a remote service document from the database.
   * 
   * This method deletes a single remote service document from the database
   * based on the provided ID.
   *
   * @param id - The unique identifier of the remote service to be removed.
   *             It should be a valid MongoDB ObjectId.
   * @returns A promise that resolves to the deleted remote service document,
   *          or null if no document matches the given ID.
   */
  async remove(id: Types.ObjectId) {
    return await this.remoteServiceSchema.findByIdAndDelete(id);
  }
}
