@Injectable()
export class FirebaseNotificationService {

    constructor() { }

    /**
     * Sends a notification to a single user using Firebase Cloud Messaging.
     * 
     * @param token - The Firebase Cloud Messaging token of the target device.
     * @param title - The title of the notification.
     * @param body - The main content of the notification.
     * @param data - Optional. Additional data to be sent with the notification.
     * 
     * @returns A Promise that resolves when the notification is sent successfully.
     *          If an error occurs, it will be caught and logged.
     */
    async sendingNotificationOneUser(token: string, title: string, body: string, data?: any) {
        const payload = {
            token,
            data,
            notification: {
                title,
                body,
                imageUrl: 'https://mechatronic-cars.s3.ca-central-1.amazonaws.com/logos/20240603/ad6cf084-168c-4922-ae56-a671094e06c1.jpeg'
            }
        }
        try {
            const response = await admin.messaging().send(payload);
            console.log('Notification Sent: ', response);
        } catch (error) {
            console.log('Notification Error: ', error);
        }
    }

    /**
     * Sends notifications to multiple users using Firebase Cloud Messaging.
     * 
     * @param tokens - An array of Firebase Cloud Messaging tokens of the target devices.
     * @param title - The title of the notification.
     * @param body - The main content of the notification.
     * @param data - Optional. Additional data to be sent with the notification.
     * 
     * @returns A Promise that resolves when the notifications are sent successfully.
     *          If an error occurs, it will be caught and logged.
     */
    async sendingNotificationMultipler(tokens: Array<string>, title: string, body: string, data?: any) {
        const payload = {
            tokens,
            data,
            notification: {
                title,
                body,
                imageUrl: 'https://mechatronic-cars.s3.ca-central-1.amazonaws.com/logos/20240603/ad6cf084-168c-4922-ae56-a671094e06c1.jpeg'
            }
        }
        try {
            const response = await admin.messaging().sendEachForMulticast(payload);
            console.log('Multiple Notification Sent: ', response);
        } catch (error) {
            console.log('Multiple Notification Error: ', error);
        }
    }
}
