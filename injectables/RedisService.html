import { HttpException, HttpStatus, Injectable } from '@nestjs/common';

// redis
import Redis from 'ioredis';
import { InjectRedis } from '@nestjs-modules/ioredis';

@Injectable()
export class RedisService {

    // redis = new Redis(config.REDIS_URL);

    constructor(
        @InjectRedis() private readonly redis: Redis
    ) { }

    /**
     * set the value for string
     * @param key 
     * @param value 
     * @returns 
     */
    async set(key: string, value: string | number) {
        return await this.redis.set(key, value);
    }

    /**
     * get value of the string
     * @param key name of string
     * @returns 
     */
    async get(key: string) {
        return await this.redis.get(key);
    }

    /**
     * increase the value of exist string
     * @param key 
     * @param value 
     * @returns 
     */
    async increase(key: string, value: string | number = 1) {
        return await this.redis.incrby(key, value);
    }

    /**
    * decrease the value of exist string
    * @param key 
    * @param value 
    * @returns 
    */
    async decrease(key: string, value: string | number = 1) {
        return await this.redis.decrby(key, value);
    }

    /**
     * set the object by JSON string function
     * @param key 
     * @param value 
     * @returns 
     */
    async setObject(key: string, value: object) {
        return await this.redis.set(key, JSON.stringify(value));
    }

    /**
     * get the value by object using JSON PARSE function
     * @param key 
     * @returns 
     */
    async getObject(key: string) {
        let result = await this.redis.get(key);
        return JSON.parse(result);
    }

    /**
     * search value from string - example filter*
     * @param key 
     * @returns 
     */
    async search(key: string) {
        return this.redis.keys(key);
    }

    /**
     * delete string from database
     * @param key 
     * @returns 
     */
    async delete(key: string) {
        return await this.redis.del(key);
    }

    /**
     * set specific object by using key from list and hash
     * @param list 
     * @param key 
     * @param value 
     */
    // list is common name to get list or hash
    async setJSONInList(list: string, key: string, value: object) {
        await this.redis.hmset(key, JSON.stringify(value))
        await this.redis.lpush(list, key)
    }

    /**
     * get specific object by using key from list and hash
     * @param list 
     * @returns 
     */
    async getJSONFromList(list: string) {
        const keys = await this.redis.lrange(list, 0, -1) // 0, -1 => all items
        const p = this.redis.pipeline()
        for (let i = 0; i < keys.length; i++) {
            const key = keys[i];
            p.hgetall(JSON.parse(key))
        }
        return await p.exec();
    }

    /**
     * scan string like find in database when database is too large
     * @param key 
     */
    async scanStream(key: string) {
        return await this.redis.scan(0, 'MATCH', key);
        const stream = this.redis.scanStream({ match: key });
        // temp array to store found keys
        let keys = [];
        // store key in temp array if found matching
        stream.on('data', (resultKeys) => {
            keys = keys.concat(resultKeys);
        });
        // return temp array after getting all matched values
        stream.on('end', () => {
            return keys;
        });
        // catch error
        stream.on('error', (error) => {
            throw new HttpException('radis scan: ' + error, HttpStatus.NOT_ACCEPTABLE);
        });
    }
}
