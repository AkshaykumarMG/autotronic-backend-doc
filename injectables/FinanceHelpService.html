import { HttpException, HttpStatus, Injectable } from '@nestjs/common';
import { CreateFinanceHelpDto } from './dto/create-finance-help.dto';
// import { UpdateFinanceHelpDto } from './dto/update-finance-help.dto';
import { ManageFinanceHelpDto } from './dto/manage-finance-help.dto';

// mongodb
import { Model, Types } from 'mongoose';
import { InjectModel } from '@nestjs/mongoose';
import { FinanceHelp, FinanceHelpStatus } from './schema/finance-help.schema';

// custom service
import { UserService } from '../user/user.service';
import { MailService } from '../../utilities/services/mail/mail.service';
import { VehicleFinanceApplicationService } from '../vehicle-finance-application/vehicle-finance-application.service';
import { AuthenticationService } from '../authentication/authentication.service';
import { FirebaseNotificationService } from '../../utilities/services/firebase/notification/firebase-notification.service';

// notification 
import { NotificationService } from '../notification/notification.service';
import { Notification_Type } from '../notification/schema/notification.schema';

// authenticate user
import { Authentication } from '../authentication/schema/authentication.schema';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';
import { UserRole } from '../user/schema/user.schema';

@Injectable()
export class FinanceHelpService {

  constructor(
    private readonly userService: UserService,
    private readonly mailService: MailService,
    private readonly notificationService: NotificationService,
    private readonly authenticationService: AuthenticationService,
    private readonly firebaseNotificationService: FirebaseNotificationService,
    private readonly vehicleFinanceApplicationSerice: VehicleFinanceApplicationService,
    @InjectModel(FinanceHelp.name) private readonly financeHelpSchema: Model<FinanceHelp>,
  ) { }

/**
 * Creates a new finance help request and notifies super admins.
 * 
 * This function performs the following steps:
 * 1. Fetches the vehicle finance application details.
 * 2. Retrieves the user information.
 * 3. Creates a new finance help entry with initial timeline.
 * 4. Finds all super admin users.
 * 5. Sends email notifications to all super admins.
 * 6. Sends push notifications and creates in-app notifications for super admins.
 * 
 * @param createFinanceHelpDto - The data transfer object containing information to create a finance help request.
 * @returns {Promise<boolean>} Returns true if the finance help request is successfully created and notifications are sent.
 * @throws {Error} Throws an error if there's a problem during the creation process or sending notifications.
 */
  async create(createFinanceHelpDto: CreateFinanceHelpDto): Promise<boolean> {

    // fetch vehicle finance application detail
    const vehicleFinanceApplication = await this.vehicleFinanceApplicationSerice.findOne(createFinanceHelpDto.vehicle_finance_application as Types.ObjectId);

    // fetch user information
    const user = await this.userService.findByID(vehicleFinanceApplication.user as Types.ObjectId);

    // add finance help data
    const createdFinanceHelp = new this.financeHelpSchema(createFinanceHelpDto);
    createdFinanceHelp.timeline = [{
      status: FinanceHelpStatus.not_assign,
      note: `${user.username} requested for finance help`,
      user: vehicleFinanceApplication.user as Types.ObjectId,
    }];
    await createdFinanceHelp.save();

    // find all super admin and send notifications
    const superAdminUsers = await this.userService.findAllBySuperAdminStatus();

    // send mail to each super admin
    for (const element of superAdminUsers) {
      // send mail to super admin
      const mailAdmin = {
        to: element.email,
        from: 'akshay@mechatronic-group.com',
        subject: 'New Message: Finance Help Request',
        templateId: 'd-6077cb411d1945e1bb519efb512ab03b',
      };
      await this.mailService.send(mailAdmin);

      // send notifications
      const pushTokens = (await this.authenticationService.findOneByEmail(element.email)).map((auth: Authentication) => { return auth.notification_token });
      if (pushTokens && pushTokens.length > 0) {
        await this.firebaseNotificationService.sendingNotificationMultipler(pushTokens, 'New Finance Help', `${user.username} requested for finance help`, { type: Notification_Type.finance_help });
        await this.notificationService.create({
          title: 'New Finance Help',
          description: `${user.username} requested for finance help`,
          from: user._id,
          to: element._id,
          type: Notification_Type.finance_help
        });
      }
    }

    return true;
  }

  /**
   * Retrieves finance help records for admin users based on their role and specified criteria.
   * 
   * This function handles different query scenarios for SuperAdmin and other admin roles.
   * It supports pagination and filtering by status.
   * 
   * @param status - The status to filter finance help records. If not provided or invalid, all statuses are included.
   * @param user - The current user object containing role information.
   * @param page - The page number for pagination (default: 0).
   * @param limit - The number of records per page (default: 10).
   * @returns An object containing the matching finance help records and the total count.
   *          The structure is { data: FinanceHelp[], count: number }.
   */
  async findAllForAdmin(status: any, user: ICurrentUser, page: number = 0, limit: number = 10) {
    const financeHelpStatus = Object.values(FinanceHelpStatus);
    status = status && financeHelpStatus.includes(status) ? { status } : {};
    if (user.role === UserRole.SuperAdmin) {
      const count = await this.financeHelpSchema.countDocuments();
      const financeHelp = await this.financeHelpSchema.find(status)
        .populate('vehicle_finance_application')
        .populate('timeline.user', { _id: 0, store: 0, favourite: 0, active: 0, createdAt: 0, updatedAt: 0 })
        .limit(limit).skip(page * limit).exec();
      return { data: financeHelp, count };
    }
    const count = await this.financeHelpSchema.countDocuments({ assign_user: user._id });
    if (Object.keys(status).length === 0) {
      const financeHelp = await this.financeHelpSchema.find({ assign_user: user._id })
        .populate('vehicle_finance_application')
        .populate('timeline.user', { _id: 0, store: 0, favourite: 0, active: 0, createdAt: 0, updatedAt: 0 })
        .limit(limit).skip(page * limit).exec();
      return { data: financeHelp, count };
    } else {
      const financeHelp = await this.financeHelpSchema.find({ assign_user: user._id, ...status })
        .populate('vehicle_finance_application')
        .populate('timeline.user', { _id: 0, store: 0, favourite: 0, active: 0, createdAt: 0, updatedAt: 0 })
        .limit(limit).skip(page * limit).exec();
      return { data: financeHelp, count };
    }
  }

  /**
   * Retrieves the count of finance help requests grouped by their status.
   * 
   * This function aggregates the finance help documents in the database,
   * grouping them by their status and calculating the count for each status.
   * 
   * @returns {Promise<Array<{ _id: string, count: number }>>} A promise that resolves to an array of objects.
   *          Each object contains:
   *          - _id: The status of the finance help requests
   *          - count: The number of finance help requests with that status
   */
  getFinanceHelpStatusWithCount(): Promise<Array<{ _id: string; count: number; }>> {
    return this.financeHelpSchema.aggregate([
      {
        $group: {
          _id: '$status',
          count: { $sum: 1 }
        }
      }
    ])
  }

  /**
   * Retrieves a single finance help record by its ID.
   * 
   * This function searches the database for a finance help document
   * matching the provided ID and returns it.
   *
   * @param id - The ObjectId of the finance help record to retrieve.
   * @returns A Promise that resolves to the found FinanceHelp document, or null if not found.
   */
  async findOne(id: Types.ObjectId) {
    return await this.financeHelpSchema.findById(id);
  }

  /**
   * Updates the status of a finance help request and manages user assignments.
   * 
   * This function performs the following actions:
   * 1. Validates the user's role for certain status changes.
   * 2. Updates the finance help status and assigns a user if applicable.
   * 3. Adds a new entry to the finance help timeline.
   * 
   * @param user - The current user making the request, decorated with @CurrentUser().
   * @param manageFinanceHelpDto - Data transfer object containing the update information.
   * @throws {HttpException} Throws an Unauthorized exception if a non-SuperAdmin user attempts to assign the finance help.
   * @returns {Promise<{ success: boolean }>} Returns an object indicating the success of the operation.
   */
  async updateFinanceStatus(@CurrentUser() user: ICurrentUser, manageFinanceHelpDto: ManageFinanceHelpDto): Promise<{ success: boolean; }> {

    // check user role and status validation for other role -  asign role only valid for super admin
    if ([FinanceHelpStatus.assign].includes(manageFinanceHelpDto.status) && user.role !== UserRole.SuperAdmin) {
      throw new HttpException('Unauthorized: Invalid Status!', HttpStatus.UNAUTHORIZED);
    }

    if ([FinanceHelpStatus.assign].includes(manageFinanceHelpDto.status) && user.role === UserRole.SuperAdmin) {
      await this.financeHelpSchema.updateOne(
        { _id: manageFinanceHelpDto.vehicle_finance_application },
        { $set: { status: FinanceHelpStatus.assign, assign_user: manageFinanceHelpDto.assign_user } })
        .exec();
    }

    // add timeline in finance app
    const timeline = {
      status: manageFinanceHelpDto.status,
      note: manageFinanceHelpDto.note,
      user: user._id,
    };
    await this.financeHelpSchema.updateOne({ _id: manageFinanceHelpDto.vehicle_finance_application }, { $push: { timeline }, status: manageFinanceHelpDto.status }).exec();

    return { success: true };
  }


  /**
   * Removes a finance help record from the database.
   * 
   * This function deletes a single finance help document from the database
   * based on the provided ObjectId.
   *
   * @param id - The ObjectId of the finance help record to be removed.
   * @returns A Promise that resolves to the deletion result. The result typically
   *          includes information such as the number of documents deleted.
   */
  async remove(id: Types.ObjectId) {
    return await this.financeHelpSchema.deleteOne({ _id: id }).exec();
  }
}
