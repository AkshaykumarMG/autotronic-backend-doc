import { Injectable, OnModuleInit, Logger, Inject } from '@nestjs/common';

// amqp lib
import { ConfirmChannel } from 'amqplib';
import amqp, { ChannelWrapper } from 'amqp-connection-manager';

// config and queue operations
import { config } from '../../../utilities/config';
import { AlertPriceService } from '../alert-price.service';
import { MailService } from '../../../utilities/services/mail/mail.service';
import { CarService } from '../../car/car.service';
import { Media } from '../../car/schema/car.schema';

/**
 * Service responsible for handling alert subscriptions through RabbitMQ.
 * It sets up a consumer to process messages from the alert subscription queue.
 */
@Injectable()
export class AlertSubscribeConsumerService implements OnModuleInit {

    private channelWrapper: ChannelWrapper;
    private readonly logger = new Logger(AlertPriceService.name);

    /**
     * Creates an instance of AlertSubscribeConsumerService.
     * @param carService - Service for car-related operations.
     * @param mailService - Service for sending emails.
     * @param alertPriceService - Service for handling price alerts.
     */
    constructor(
        private readonly carService: CarService,
        private readonly mailService: MailService,
        private readonly alertPriceService: AlertPriceService,
    ) {
        const connection = amqp.connect([config.RABBITMQ_CAR_URL]);
        this.channelWrapper = connection.createChannel();
    }

    /**
     * Initializes the module by setting up the RabbitMQ consumer.
     * This method is called automatically when the module is initialized.
     * It sets up the channel, asserts the queue, and starts consuming messages.
     * For each received message, it processes the alert subscription and sends emails to subscribers.
     * 
     * @returns {Promise<void>} A promise that resolves when the initialization is complete.
     * @throws {Error} If there's an error during the initialization process.
     */
    public async onModuleInit(): Promise<void> {
        try {
            await this.channelWrapper.addSetup(async (channel: ConfirmChannel) => {

                await channel.assertQueue(config.ALERT_SUBCRIBE_QUEUE_NAME, { durable: true });

                await channel.consume(config.ALERT_SUBCRIBE_QUEUE_NAME, async (message) => {
                    if (message) {
                        const response = JSON.parse(message.content.toString());
                        // this.logger.log('Received message:', response); 

                        // find all records who subscribe price alert for this car
                        const priceAlertCars = await this.alertPriceService.findAllRecordByCarID(response.id);
                        // get emails by finded users
                        const emails = priceAlertCars.map(car => car.email);
                        // find car details
                        let car: any = await this.carService.findOneCarByID(response.id);
                        car = JSON.parse(JSON.stringify(car));

                        // send mail to users
                        const mailClient = {
                            to: emails,
                            from: 'akshay@mechatronic-group.com', // Fill it with your validated email on SendGrid account
                            templateId: 'd-c523ad762f2745d19c63daf41cf52817',
                            dynamicTemplateData: {
                                title: car.name,
                                location: car.seller_type === 'dealer' ? `${car.store.address.city}, ${car.store.address.state}` : `${car.user.address.city}, ${car.user.address.state}`,
                                price: car.asking_price,
                                odometer: car.odometer,
                                image: car.images.sort((a: Media, b: Media) => a.position > b.position ? 1 : -1)[0].urlThumbnail,
                                link: `${config.FRONTEND_URL}detail/${car._id}`
                            },
                            hideWarnings: true, // now the warning won't be logged
                        };
                        await this.mailService.send(mailClient);

                        // inform queue work is done
                        channel.ack(message);
                    }
                });
            });
            this.logger.log('Alert Subscribe Consumer service started...');
        } catch (err) {
            this.logger.error('Error starting the consumer:', err);
        }
    }
}
