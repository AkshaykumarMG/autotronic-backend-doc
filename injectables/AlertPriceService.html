import { forwardRef, HttpException, HttpStatus, Inject, Injectable } from '@nestjs/common';
import { CreateAlertPriceDto } from './dto/create-alert-price.dto';
// import { UpdateAlertPriceDto } from './dto/update-alert-price.dto';

// mail service
import { MailService } from '../../utilities/services/mail/mail.service';

// car service
import { CarService } from '../car/car.service';

// mongoose
import { Model, Types } from 'mongoose';
import { InjectModel } from '@nestjs/mongoose';
import { AlertPrice } from './schema/alert-price.schema';
import { Media } from '../car/schema/car.schema';

// config
import { config } from '../../utilities/config';

// queue add & remove media parallel, Manage filter operations(increment, decrement)
import amqp, { ChannelWrapper } from 'amqp-connection-manager';
import { Channel } from 'amqplib';

@Injectable()
export class AlertPriceService {

  private readonly channelWrapper: ChannelWrapper;

  constructor(
    @Inject(forwardRef(() => CarService))
    private readonly carService: CarService,
    private readonly mailService: MailService,
    @InjectModel(AlertPrice.name) private readonly alertPriceSchema: Model<AlertPrice>
  ) {
    const connection = amqp.connect([config.RABBITMQ_CAR_URL]);
    this.channelWrapper = connection.createChannel({
      setup: (channel: Channel) => {
        return channel.assertQueue(config.ALERT_SUBCRIBE_QUEUE_NAME, { durable: true });
      },
    });
  }

  /**
   * Creates a new alert price entry and sends an email notification.
   * 
   * This function performs the following steps:
   * 1. Saves the new alert price to the database.
   * 2. Retrieves the associated car details.
   * 3. Prepares and sends an email notification with the car details.
   * 
   * @param createAlertPriceDto - The data transfer object containing the alert price details.
   * @param createAlertPriceDto.email - The email address of the recipient.
   * @param createAlertPriceDto.car - The ObjectId of the associated car.
   * 
   * @returns {Promise<void>} A promise that resolves when the alert price is created and the email is sent.
   * 
   * @throws {Error} If there's an issue saving to the database or sending the email.
   */
  async create(createAlertPriceDto: CreateAlertPriceDto): Promise<void> {

    // add in db
    const createdAward = new this.alertPriceSchema(createAlertPriceDto);
    await createdAward.save();

    // get car detail
    const car: any = await this.carService.findOneCarForUpdateOrViewByID(createAlertPriceDto.car as Types.ObjectId);

    // mail service
    const mailClient = {
      to: createAlertPriceDto.email,
      from: 'akshay@mechatronic-group.com', // Fill it with your validated email on SendGrid account
      templateId: 'd-ddad62db50524f3b871ada4149a0a287',
      dynamicTemplateData: {
        title: car.name,
        price: car.asking_price,
        odometer: car.odometer,
        location: car.seller_type === 'dealer' ? `${car.store.address.city}, ${car.store.address.state}` : `${car.user.address.city}, ${car.user.address.state}`,
        image: car.images.sort((a: Media, b: Media) => a.position > b.position ? 1 : -1)[0].url,
        link: `${config.FRONTEND_URL}detail/${car._id}`
      },
      hideWarnings: true, // now the warning won't be logged
    };

    await this.mailService.send(mailClient);
  }

  /**
   * Sends a price alert to a RabbitMQ queue and retrieves all alert records for a specific car.
   * 
   * This function performs two main tasks:
   * 1. It sends a message to a RabbitMQ queue with the car's ID for price alert processing.
   * 2. It retrieves all alert records associated with the given car ID from the database.
   * 
   * @param id - The ObjectId of the car for which the price alert is being sent.
   * 
   * @returns A Promise that resolves to an array of AlertPrice documents associated with the car,
   *          sorted in descending order by creation date.
   * 
   * @throws {HttpException} If there's an error while sending the message to the queue.
   */
  async sendPriceAlertToQueue(id: Types.ObjectId) {
    try {
      await this.channelWrapper.sendToQueue(config.ALERT_SUBCRIBE_QUEUE_NAME,
        Buffer.from(JSON.stringify({ id })),
        {
          persistent: true,
        },
      );
    } catch (error) {
      throw new HttpException(
        'Error adding during price alert, Kindly try again',
        HttpStatus.INTERNAL_SERVER_ERROR,
      );
    }
  
    return await this.alertPriceSchema.find({ car: id })
      .sort({ createdAt: -1 })
      .exec();
  }

  /**
   * Retrieves all alert price records associated with a specific car.
   *
   * This function queries the database for all AlertPrice documents that are linked
   * to the specified car ID. The results are sorted in descending order by creation date.
   *
   * @param id - The unique identifier of the car (typically a MongoDB ObjectId as a string).
   * 
   * @returns A Promise that resolves to an array of AlertPrice documents associated with the specified car,
   *          sorted in descending order by their creation date.
   */
  async findAllRecordByCarID(id: string) {
    return await this.alertPriceSchema.find({ car: id })
      .sort({ createdAt: -1 })
      .exec();
  }

  /**
   * Retrieves all alert price records for a specific user's email, along with associated car details.
   * 
   * This function performs an aggregation pipeline to:
   * 1. Match all records with the given email.
   * 2. Lookup associated car details, including the first image, name, VIN, ID, and asking price.
   * 3. Project out unnecessary fields from the result.
   * 4. Sort the results by creation date in descending order.
   *
   * @param email - The email address of the user for whom to retrieve alert prices.
   * @returns A Promise that resolves to an array of alert price documents with associated car details,
   *          sorted by creation date in descending order.
   */
  async findAllByUserEmail(email: string) {
    return await this.alertPriceSchema.aggregate([
      { $match: { email } },
      {
        $lookup: {
          from: "cars",
          localField: "car",
          foreignField: "_id",
          as: "car",
          pipeline: [
            { $unwind: '$images' },
            { $sort: { 'images.position': 1 } },
            { $project: { 'images.url': 1, name: 1, vin_number: 1, _id: 1, asking_price: 1 } },
            { $limit: 1 }
          ],
        },
      },
      {
        $project: {
          category: 0,
          active: 0,
          createdAt: 0,
          updatedAt: 0
        }
      }
    ])
      .sort({ createdAt: -1 })
      .exec();
  }

  /**
   * Removes an alert price record from the database.
   * 
   * This function deletes the alert price document that matches the provided ID.
   * 
   * @param id - The unique identifier of the alert price record to be removed.
   * @returns A Promise that resolves to the deleted AlertPrice document, or null if no document was found.
   */
  async remove(id: Types.ObjectId) {
    return this.alertPriceSchema.findByIdAndDelete(id);
  }
}
