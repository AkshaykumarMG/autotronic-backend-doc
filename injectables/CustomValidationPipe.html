import { ArgumentMetadata, Injectable, PipeTransform } from '@nestjs/common';
import { validate } from 'class-validator';
import { plainToInstance } from 'class-transformer';

@Injectable()
export class CustomValidationPipe implements PipeTransform {

  /**
   * Transforms and validates the input value based on the provided metadata.
   * 
   * This method performs the following steps:
   * 1. Checks if validation is required for the given metatype.
   * 2. Transforms the plain object to an instance of the metatype.
   * 3. Validates the transformed object.
   * 4. Returns an error object if validation fails, otherwise returns the original value.
   *
   * @param value - The input value to be transformed and validated.
   * @param param1 - An object containing the ArgumentMetadata.
   * @param param1.metatype - The metatype of the argument, used for validation.
   * @returns The original value if validation passes, or an error object if validation fails.
   *          The error object includes the current timestamp and an error message.
   */
  async transform(value: any, { metatype }: ArgumentMetadata) {
    if (!metatype || !this.toValidate(metatype)) {
      return value;
    }
    const object = plainToInstance(metatype, value);
    const errors = await validate(object);
    if (errors.length > 0) {
      return {
        time: new Date(),
        // method: ,
        message: 'Validation Failed: ' + errors
      }
      // throw new BadRequestException('Validation failed');
    }
    return value;
  }

  /**
   * Determines if a given metatype should be validated.
   * 
   * This method checks if the provided metatype is a custom class that requires validation,
   * as opposed to a built-in JavaScript type.
   *
   * @param metatype - The constructor function of the type to be checked.
   * @returns A boolean indicating whether the metatype should be validated (true) or not (false).
   */
  private toValidate(metatype: Function): boolean {
    const types: Function[] = [String, Boolean, Number, Array, Object];
    return !types.includes(metatype);
  }
}
