import { Injectable, OnModuleInit, Logger, Inject, forwardRef } from '@nestjs/common';

// amqp lib
import { ConfirmChannel } from 'amqplib';
import amqp, { ChannelWrapper } from 'amqp-connection-manager';

// config and queue operations
import { config } from '../../../utilities/config';

// used service
import { CarService } from '../car.service';
import { FilterService } from '../../filter/filter.service';
import { AwsS3Service } from '../../../utilities/services/aws_s3/aws_s3.service';
import { RedisService } from '../../../utilities/services/redis/redis.service';

// relationship db 
import { Store } from '../../store/schema/store.schema';
import { Filter, labelValue } from '../../filter/schema/filter.schema';
import { Media } from '../schema/car.schema';
import { UserService } from '../../user/user.service';
import { UserRole } from '../../user/schema/user.schema';
import { AlertPriceService } from '../../alert-price/alert-price.service';

// mail service
import { MailService } from '../../../utilities/services/mail/mail.service';

// mongoose
import { InjectConnection } from '@nestjs/mongoose';
import * as mongoose from 'mongoose';

interface GetResponse {
    mode: string,
    data: any
}

@Injectable()
export class CarConsumerService implements OnModuleInit {

    private readonly channelWrapper: ChannelWrapper;
    private readonly logger = new Logger(CarConsumerService.name);

    constructor(
        private readonly carService: CarService,
        private readonly userService: UserService,
        private readonly mailService: MailService,
        private readonly awsS3Service: AwsS3Service,
        private readonly filterService: FilterService,
        @Inject(forwardRef(() => AlertPriceService))
        private readonly alertPriceService: AlertPriceService,
        @Inject(RedisService) private readonly redisService: RedisService,
        @InjectConnection() private readonly connectionDB: mongoose.Connection
    ) {
        const connectionQueue = amqp.connect([config.RABBITMQ_CAR_URL]);
        this.channelWrapper = connectionQueue.createChannel();
    }

    public async onModuleInit() {
        try {
            await this.channelWrapper.addSetup(async (channel: ConfirmChannel) => {

                await channel.assertQueue(config.QUEUE_ADVERTISEMENT_QUEUE_NAME, { durable: true });

                await channel.consume(config.QUEUE_ADVERTISEMENT_QUEUE_NAME, async (message) => {
                    if (message) {
                        const response: GetResponse = JSON.parse(message.content.toString());
                        // this.logger.log('Received message:', response.mode);
                        // add advertisement
                        if (response.mode === config.QUEUE_ADVERTISEMENT_TOPIC_ADD_ADVERTISEMENT) {
                            const transactionSession = await this.connectionDB.startSession();
                            transactionSession.startTransaction();
                            try {
                                // get data from payload
                                const createCar = response.data;

                                // get filters to set title of the advertisement
                                const filters = await this.filterService.findAllForAdmin();
                                const make = (filters?.find((filter: Filter) => filter.name.value === 'make').data as labelValue[]).find((element: any) => element.value === createCar.make).label;
                                const model = (filters?.find((filter: any) => filter.name.value === 'model').data as labelValue[]).find((element: any) => element.value === createCar.model).label;
                                const color = (filters?.find((filter: any) => filter.name.value === 'color').data as labelValue[]).find((element: any) => element.value === createCar.color).label;
                                const transmission = (filters?.find((filter: any) => filter.name.value === 'transmission').data as labelValue[]).find((element: any) => element.value === createCar.transmission).label;
                                let trim = (filters?.find((filter: any) => filter.name.value === 'trim').data as labelValue[]).find((element: any) => element.value === createCar.trim).label;
                                if(trim === 'Unknown') {
                                    trim = '';
                                }
                                createCar.name = `${createCar.year} ${make} ${model} ${color} ${transmission} ${trim}`;

                                // add car in database
                                const carAdded = await (await this.carService.createByQueue(createCar)).populate('store');

                                // get user info
                                const user = await this.userService.findByID(createCar.createdBy);
                                // save car location by user role
                                if (user.role == UserRole.Dealer || user.role === UserRole.Admin || user.role === UserRole.SuperAdmin) {
                                    // update location of car & get car details by id
                                    const car = await this.carService.refCarSchema().findById(carAdded._id).populate('store');
                                    const store = car.store as Store;
                                    carAdded.location = { type: 'Point', coordinates: [store.location.coordinates[0], store.location.coordinates[1]] as any };
                                } else {
                                    carAdded.location = user.location;
                                }
                                carAdded.save();

                                // get all filters - and increment by 1
                                await Promise.allSettled(filters.map(async (filter: Filter) => {
                                    if (filter.name.value in createCar && (filter.type === 'checkbox' || filter.type === 'textbox-list')) {
                                        return await this.filterService.updateFilterCountOnAdd(filter._id, createCar[filter.name.value]);
                                    }
                                    if (filter.name.value in createCar && (filter.type === 'multi-select' || filter.type === 'multi-choice')) {
                                        createCar[filter.name.value].forEach(async (element: labelValue) => {
                                            return await this.filterService.updateFilterCountOnAdd(filter._id, element.value as string);
                                        })
                                    }
                                }));

                                // send mail to user
                                const mailClient = {
                                    to: user.email,
                                    from: 'akshay@mechatronic-group.com', // Fill it with your validated email on SendGrid account
                                    templateId: 'd-a760728bd92f4ad5a60603d55fe748d5',
                                    dynamicTemplateData: {
                                        make, model, vin: carAdded.vin_number,
                                        link: `${config.FRONTEND_URL}/user/car`,
                                        title: carAdded.name,
                                        store: (carAdded.store as Store).name
                                    },
                                    sendAt: Math.floor(+new Date() / 1000) + 10
                                };
                                // await this.mailService.send(mailClient);

                                // complete transaction
                                await transactionSession.commitTransaction();
                            } catch (error) {
                                console.log('consumer add advertisement error', JSON.stringify(error));
                                await transactionSession.abortTransaction();
                            } finally {
                                await transactionSession.endSession();
                                channel.ack(message);
                            }
                        } else if (response.mode === config.QUEUE_ADVERTISEMENT_TOPIC_UPDATE_ADVERTISEMENT) {
                            const transactionSession = await this.connectionDB.startSession();
                            transactionSession.startTransaction();
                            try {
                                // get car details by id
                                const updateCarDto = response.data;
                                let car = await this.carService.refCarSchema().findById(updateCarDto._id);

                                // find all filters - decreased by 1
                                const filters = await this.filterService.findAllForAdmin();
                                // convert in object from bson
                                car = JSON.parse(JSON.stringify(car));
                                await Promise.allSettled(filters.map(async (filter: Filter) => {
                                    if (filter.name.value in car && (filter.type === 'checkbox' || filter.type === 'textbox-list')) {
                                        if (car[filter.name.value] !== updateCarDto[filter.name.value]) {
                                            // decrease by old filter option
                                            await this.filterService.updateFilterCountOnRemove(filter._id, car[filter.name.value]);
                                            // increase by new filter option
                                            await this.filterService.updateFilterCountOnAdd(filter._id, updateCarDto[filter.name.value]);
                                        }
                                    }
                                    if (filter.name.value in car && filter.type === 'multi-select') {
                                        // check is the any chnages in multi-sleect array
                                        if (JSON.stringify(car[filter.name.value]) !== JSON.stringify(updateCarDto[filter.name.value])) {
                                            car[filter.name.value].forEach(async (element: labelValue) => {
                                                await this.filterService.updateFilterCountOnRemove(filter._id, element.value as string);
                                            });
                                            updateCarDto[filter.name.value].forEach(async (element: labelValue) => {
                                                await this.filterService.updateFilterCountOnAdd(filter._id, element.value as string);
                                            });
                                        }
                                    }
                                }));


                                // update title of the advertisement
                                const make = (filters?.find((filter: Filter) => filter.name.value === 'make').data as labelValue[]).find((element: any) => element.value === updateCarDto.make).label;
                                const model = (filters?.find((filter: any) => filter.name.value === 'model').data as labelValue[]).find((element: any) => element.value === updateCarDto.model).label;
                                const color = (filters?.find((filter: any) => filter.name.value === 'color').data as labelValue[]).find((element: any) => element.value === updateCarDto.color).label;
                                const transmission = (filters?.find((filter: any) => filter.name.value === 'transmission').data as labelValue[]).find((element: any) => element.value === updateCarDto.transmission).label;
                                const trim = (filters?.find((filter: any) => filter.name.value === 'trim').data as labelValue[]).find((element: any) => element.value === updateCarDto.trim).label;
                                updateCarDto.name = `${updateCarDto.year} ${make} ${model} ${color} ${transmission} ${trim}`;

                                // update car in db
                                const updatedCar = await (await this.carService.update(car._id, updateCarDto)).populate('store');

                                // check price - if not match then add in price history
                                if (updateCarDto.asking_price !== car.asking_price) {
                                    if (!updatedCar.price_history || updatedCar.price_history.length == 0) {
                                        updatedCar.price_history = [updatedCar.asking_price, updateCarDto.asking_price];
                                    } else {
                                        updatedCar.price_history = [...updatedCar.price_history, updateCarDto.asking_price];
                                        // console.log('send price queue from car: ', updatedCar._id);
                                        // send price alert mail to all user who subscribe price alert for this advertisement
                                        await this.alertPriceService.sendPriceAlertToQueue(updatedCar._id);
                                    }
                                }

                                // check store or location is changed
                                if (!car.location || updateCarDto.store != String(car.store._id)) {
                                    updatedCar.location = { type: 'Point', coordinates: [(updatedCar.store as Store).location.coordinates[0], (updatedCar.store as Store).location.coordinates[1]] };
                                }

                                // save updated changes in db
                                updatedCar.save();

                                // console.log('update car: ', updateCarDto.store, updatedCar);
                                await transactionSession.commitTransaction();
                            } catch (error) {
                                console.log('consumer error', JSON.stringify(error));
                                await transactionSession.abortTransaction();
                            } finally {
                                await transactionSession.endSession();
                                channel.ack(message);
                            }
                        } else if (response.mode === config.QUEUE_ADVERTISEMENT_TOPIC_ADD_REMOVE_MEDIA) {
                            const transactionSession = await this.connectionDB.startSession();
                            transactionSession.startTransaction();
                            try {
                                const createCarMediaDto = response.data;
                                if (createCarMediaDto.mode === 'add') {
                                    // add the media here
                                    await this.carService.refCarSchema().findOneAndUpdate({ stock_number: createCarMediaDto.id }, { $push: { [createCarMediaDto.field]: createCarMediaDto.media } });
                                }
                                if (createCarMediaDto.mode === 'delete') {
                                    // remove the media here
                                    await this.carService.refCarSchema().findByIdAndUpdate(createCarMediaDto.id, { $pull: { [createCarMediaDto.field]: { name: createCarMediaDto.media.name } } });
                                    await this.awsS3Service.delete(createCarMediaDto.media.path + createCarMediaDto.media.name);
                                }
                                transactionSession.commitTransaction();
                            } catch (error) {
                                console.log('consumer manage media error', JSON.stringify(error));
                                await transactionSession.abortTransaction();
                            } finally {
                                await transactionSession.endSession();
                                channel.ack(message);
                            }
                        } else if (response.mode === config.QUEUE_ADVERTISEMENT_TOPIC_UPDATE_POSITION_ADVERTISEMENT) {
                            const transactionSession = await this.connectionDB.startSession();
                            transactionSession.startTransaction();
                            try {
                                const updateCarMediaPositionDto = response.data;
                                // update media postion
                                await this.carService.updateCarMediaPosition(updateCarMediaPositionDto);
                                transactionSession.commitTransaction();
                            } catch (error) {
                                console.log('consumer error', JSON.stringify(error));
                                await transactionSession.abortTransaction();
                            } finally {
                                await transactionSession.endSession();
                                channel.ack(message);
                            }
                        } else if (response.mode === config.QUEUE_ADVERTISEMENT_TOPIC_REMOVE_ADVERTISEMENT) {
                            const transactionSession = await this.connectionDB.startSession();
                            transactionSession.startTransaction();
                            try {
                                const data = response.data;
                                // get car details by id
                                let car = await this.carService.refCarSchema().findById(data.id);

                                //remove media files
                                // images
                                await Promise.allSettled(car.images.map(async (element: Media) => {
                                    await this.awsS3Service.delete(element.path + element.name)
                                }));
                                // videos
                                if (car.videos.length > 0) {
                                    await Promise.allSettled(car.videos.map(async (element: Media) => {
                                        await this.awsS3Service.delete(element.path + element.name)
                                    }));
                                }
                                // deactive from list - not deleting from db
                                // await this.carService.refCarSchema().findByIdAndUpdate(data.id, { $set: { active: false } });
                                await this.carService.refCarSchema().findByIdAndDelete(data.id);

                                // decrease car count in redis
                                const getCarRedis = await this.redisService.get('all:car');
                                if (getCarRedis) {
                                    await this.redisService.decrease('all:car');
                                }

                                // find all filters - decreased by 1
                                const filters = await this.filterService.findAllForAdmin();
                                // convert in object from bson
                                car = JSON.parse(JSON.stringify(car));
                                await Promise.allSettled(filters.map(async (filter: Filter) => {
                                    if (filter.name.value in car && (filter.type === 'checkbox' || filter.type === 'textbox-list')) {
                                        const updatedFilter = await this.filterService.updateFilterCountOnRemove(filter._id, car[filter.name.value]);
                                        return updatedFilter;
                                    }
                                    if (filter.name.value in car && filter.type === 'multi-select') {
                                        car[filter.name.value].forEach(async (element: labelValue) => {
                                            const updatedFilter = await this.filterService.updateFilterCountOnRemove(filter._id, element.value as string);
                                            return updatedFilter;
                                        })
                                    }
                                }));
                                await transactionSession.commitTransaction();
                            } catch (error) {
                                console.log('consumer error', JSON.stringify(error));
                                await transactionSession.commitTransaction();
                            } finally {
                                await transactionSession.endSession();
                                channel.ack(message);
                            }
                        } else {
                            // not match data operation send empty response and remove message from queue
                            channel.ack(message);
                        }
                    }
                });
            });
            this.logger.log('Consumer service started and listening for messages.');
        } catch (err) {
            this.logger.error('Error starting the consumer:', err);
        }
    }
}
