import { Controller, Get, Post, Body, Patch, Param, Delete, Query, UseGuards } from '@nestjs/common';
import { RemoteServiceService } from './remote-service.service';
import { CreateRemoteServiceDto } from './dto/create-remote-service.dto';
import { UpdateRemoteServiceDto } from './dto/update-remote-service.dto';

// mongoose
import { Types } from 'mongoose';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';

// swagger cookie
import { ApiCookieAuth, ApiParam } from '@nestjs/swagger';

@Controller('remote-service')
@ApiCookieAuth()
@UseGuards(AuthGuard, RoleGuard)
@Roles(UserRole.SuperAdmin)
export class RemoteServiceController {
  constructor(private readonly remoteServiceService: RemoteServiceService) {}

  /**
   * Creates a new remote service entry.
   * 
   * This function handles the creation of a new remote service. It sets the
   * 'createdBy' field of the DTO to the current user's ID before passing it
   * to the service layer for creation.
   *
   * @param {ICurrentUser} user - The current authenticated user.
   * @param {CreateRemoteServiceDto} createRemoteServiceDto - The DTO containing the data for creating a new remote service.
   * @returns {Promise<any>} A promise that resolves to the created remote service entry.
   */
  @Post()
  async create(@CurrentUser() user: ICurrentUser, @Body() createRemoteServiceDto: CreateRemoteServiceDto): Promise<any> {
    createRemoteServiceDto.createdBy = user._id;
    return await this.remoteServiceService.create(createRemoteServiceDto);
  }

  /**
   * Retrieves all remote service entries with pagination.
   * 
   * This endpoint fetches a paginated list of all remote service entries.
   * It uses query parameters to determine the page number and the number
   * of items per page.
   *
   * @param {number} page - The page number of results to return. Used for pagination.
   * @param {number} limit - The maximum number of items to return per page.
   * @returns {Promise<any>} A promise that resolves to an object containing the paginated list of remote service entries.
   *                         The exact structure of the return object depends on the implementation of the service method.
   */
  @Get()
  async findAll(@Query('page') page: number, @Query('limit') limit: number): Promise<any> {
    return await this.remoteServiceService.findAll(page, limit);
  }

  /**
   * Retrieves a list of remote services for selection purposes.
   * 
   * This endpoint is accessible to users with Dealer or Person roles. It fetches
   * a list of remote services, likely in a simplified format suitable for
   * populating selection dropdowns or similar UI elements.
   *
   * @returns {Promise<any>} A promise that resolves to a list of remote services.
   *                         The exact structure of the returned data depends on
   *                         the implementation of the service method.
   */
  @Get('select')
  @Roles(UserRole.Dealer, UserRole.Person)
  async findAllForSelect(): Promise<any> {
    return await this.remoteServiceService.findAllForSelection();
  }

  /**
   * Retrieves a single remote service entry by its ID.
   * 
   * This endpoint fetches the details of a specific remote service entry
   * based on the provided ID. It's useful for getting comprehensive
   * information about a particular remote service.
   *
   * @param {Types.ObjectId} id - The unique identifier of the remote service to retrieve.
   *                              This should be a valid MongoDB ObjectId.
   * @returns {Promise<any>} A promise that resolves to the remote service entry matching the provided ID.
   *                         The exact structure of the returned object depends on the
   *                         implementation of the service method.
   */
  @Get(':id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  async findOne(@Param('id') id: Types.ObjectId): Promise<any> {
    return await this.remoteServiceService.findOne(id);
  }

  /**
   * Updates an existing remote service entry.
   * 
   * This endpoint allows for the modification of a specific remote service entry
   * identified by its unique ID. It accepts partial updates, meaning only the
   * fields provided in the DTO will be updated.
   *
   * @param {Types.ObjectId} id - The unique identifier of the remote service to update.
   *                              This should be a valid MongoDB ObjectId.
   * @param {UpdateRemoteServiceDto} updateRemoteServiceDto - The DTO containing the fields to be updated.
   *                                                         Only the fields provided will be modified.
   * @returns {Promise<any>} A promise that resolves to the updated remote service entry.
   *                         The returned object represents the remote service after the update operation.
   */
  @Patch(':id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  async update(@Param('id') id: Types.ObjectId, @Body() updateRemoteServiceDto: UpdateRemoteServiceDto): Promise<any> {
    return await this.remoteServiceService.update(id, updateRemoteServiceDto);
  }

  /**
   * Deletes a remote service entry by its ID.
   * 
   * This function removes a specific remote service entry from the database
   * based on the provided ID. It's a DELETE operation that doesn't return
   * any data, but instead confirms the successful deletion by returning
   * a JSON object with a 'success' property set to true.
   *
   * @param {Types.ObjectId} id - The unique identifier of the remote service to delete.
   *                              This should be a valid MongoDB ObjectId.
   * @returns {Promise<any>} A promise that resolves to a JSON object with a 'success' property set to true.
   *                         This indicates that the remote service was successfully deleted.
   */
  @Delete(':id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  async remove(@Param('id') id: Types.ObjectId): Promise<any> {
    await this.remoteServiceService.remove(id);
    return { success: true };
  }
}
