import { Controller, Get, Post, Body, Patch, Param, Delete, UseGuards, HttpException, HttpStatus, Query } from '@nestjs/common';
import { StoreService } from './store.service';
import { CreateStoreDto } from './dto/create-store.dto';
import { UpdateStoreDto } from './dto/update-store.dto';

// mongoose
import { Types } from 'mongoose';
import { Store, StoreStatus } from './schema/store.schema';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';
import { SearchRegionStatusDto } from './dto/search-region-status.dto';
import { ManageUserStatusDto } from './dto/manage-user-status.dto';

// swagger cookie
import { ApiCookieAuth, ApiParam } from '@nestjs/swagger';

@Controller('store')
// @UseGuards(AuthGuard, RoleGuard)
export class StoreController {
  constructor(private readonly storeService: StoreService) { }

  /**
   * Creates a new store for the authenticated dealer.
   * 
   * This endpoint is protected by authentication and role-based guards, 
   * allowing only users with the Dealer role to access it.
   *
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   * @param {CreateStoreDto} createStoreDto - The DTO containing the store creation data.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success of the operation.
   */
  @Post()
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer)
  async create(@CurrentUser() user: ICurrentUser, @Body() createStoreDto: CreateStoreDto): Promise<{ success: boolean; }> {
    createStoreDto.user = user._id;
    await this.storeService.create(createStoreDto);
    return { success: true };
  }

  /**
   * Finds stores by region and status for a super admin user.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with the SuperAdmin role to access it.
   *
   * @param {SearchRegionStatusDto} searchRegionStatusDto - The DTO containing search criteria for region and status.
   * @returns {Promise<Store[]>} A promise that resolves to an array of Store objects matching the search criteria.
   */
  @Post('status/region')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async findByRegionAndStatus(@Body() searchRegionStatusDto: SearchRegionStatusDto): Promise<Store[]> {
    return await this.storeService.findAllForUserByRegionAndStatus(searchRegionStatusDto);
  }

  /**
   * Sets the status of a store by an admin or super admin user.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with SuperAdmin or Admin roles to access it.
   *
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   * @param {ManageUserStatusDto} manageUserStatusDto - The DTO containing the store status management data.
   * @returns {Promise<any>} A promise that resolves to the result of managing the store status for the user.
   */
  @Post('admin/status')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin, UserRole.Admin)
  async setStatusByAdmin(@CurrentUser() user: ICurrentUser, @Body() manageUserStatusDto: ManageUserStatusDto): Promise<any> {
    return await this.storeService.manageStoreStatusForUser(user, manageUserStatusDto);
  }

  /**
   * Sets the status of a store by a dealer user.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with the Dealer role to access it.
   *
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   * @param {ManageUserStatusDto} manageUserStatusDto - The DTO containing the store status management data.
   * @returns {Promise<any>} A promise that resolves to the result of managing the store status for the admin.
   */
  @Post('dealer/status')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer)
  async setStatusByDealer(@CurrentUser() user: ICurrentUser, @Body() manageUserStatusDto: ManageUserStatusDto): Promise<any> {
    return await this.storeService.manageStoreStatusForAdmin(user, manageUserStatusDto);
  }

  /**
   * Retrieves all stores for the authenticated user based on their role.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing access to users with Dealer, Admin, or SuperAdmin roles.
   *
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   * @returns {Promise<Store[]>} A promise that resolves to an array of Store objects associated with the user.
   */
  @Get()
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer, UserRole.Admin, UserRole.SuperAdmin)
  async findAllForUser(@CurrentUser() user: ICurrentUser): Promise<Store[]> {
    return await this.storeService.findAllForUser(user);
  }

  /**
   * Retrieves the count of stores for each status.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with Admin or SuperAdmin roles to access it.
   * 
   * @returns {Promise<{ [key: string]: number }>} A promise that resolves to an object where keys are store statuses and values are the count of stores with that status.
   */
  @Get('status')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Admin, UserRole.SuperAdmin)
  async findAllStoreStatusWithCount() {
    return await this.storeService.getStoreStatusWithCount();
  }

  /**
   * Retrieves a paginated list of stores for selection purposes.
   * 
   * This endpoint is protected by authentication and role-based guards.
   * It allows fetching stores in a paginated manner, which can be useful
   * for populating select dropdowns or similar UI components.
   *
   * @param {number} page - The page number of results to retrieve. Used for pagination.
   * @param {number} limit - The maximum number of stores to return per page.
   * @returns {Promise<any>} A promise that resolves to an object containing the paginated store data.
   *                         The exact structure depends on the implementation of storeService.findAllForSelect().
   */
  @Get('select')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  async findAllForSelect(@Query('page') page: number, @Query('limit') limit: number): Promise<any> {
    return await this.storeService.findAllForSelect(page, limit);
  }

  /**
   * Retrieves a paginated list of stores for admin users based on the specified status.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with Admin or SuperAdmin roles to access it.
   *
   * @param {StoreStatus} status - The status of stores to retrieve. This is an enum value from StoreStatus.
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   * @param {number} page - The page number of results to retrieve. Used for pagination.
   * @param {number} limit - The maximum number of stores to return per page.
   * @returns {Promise<any>} A promise that resolves to an object containing the paginated store data.
   *                         The exact structure depends on the implementation of storeService.findAllForAdmin().
   */
  @Get('admin/:status')
  @ApiCookieAuth()
  @ApiParam({name: 'status', required: true, enum: StoreStatus })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Admin, UserRole.SuperAdmin)
  async findAllStoreForAdmin(@Param('status') status: StoreStatus, @CurrentUser() user: ICurrentUser, @Query('page') page: number, @Query('limit') limit: number): Promise<any> {
    return await this.storeService.findAllForAdmin(status, user, page, limit);
  }

  /**
   * Retrieves all stores associated with the authenticated dealer user.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with the Dealer role to access it.
   *
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   *                              Contains user information, including the user's ID.
   * @returns {Promise<Store[]>} A promise that resolves to an array of Store objects associated with the dealer.
   */
  @Get('dealer')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer)
  async findStoreForDealer(@CurrentUser() user: ICurrentUser): Promise<Store[]> {
    return await this.storeService.findAllForDealer(user._id);
  }

  /**
   * Retrieves a single store by its ID.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with the Dealer role to access it.
   *
   * @param {Types.ObjectId} id - The unique identifier of the store to retrieve.
   * @returns {Promise<Store>} A promise that resolves to the Store object matching the provided ID.
   *                           If no store is found, the behavior depends on the implementation of storeService.findOne().
   */
  @Get(':id')
  @ApiCookieAuth()
  @ApiParam({name: 'id', required: true, type: 'string' })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer)
  async findOne(@Param('id') id: Types.ObjectId): Promise<Store> {
    return await this.storeService.findOne(id);
  }

  /**
   * Retrieves store information based on the provided alias.
   * 
   * This endpoint allows fetching store details using a unique alias identifier.
   * It's not protected by authentication, allowing public access to store information.
   *
   * @param {string} alias - The unique alias of the store to retrieve.
   *                         This is typically a human-readable identifier for the store.
   * @returns {Promise<Store>} A promise that resolves to the Store object matching the provided alias.
   *                           If no store is found, the behavior depends on the implementation of storeService.findOneByAlias().
   */
  @Get('info/:alias')
  @ApiParam({name: 'alias', required: true, type: 'string' })
  async findStoreByAlias(@Param('alias') alias: string): Promise<Store> {
    return await this.storeService.findOneByAlias(alias);
  }

  /**
   * Updates a store's information.
   * 
   * This endpoint is protected by authentication and role-based guards,
   * allowing only users with the Dealer role to access it. It ensures that
   * only the store owner can update their store's information.
   *
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   *                              Contains user information, including the user's ID.
   * @param {Types.ObjectId} id - The unique identifier of the store to be updated.
   * @param {UpdateStoreDto} updateStoreDto - The DTO containing the updated store information.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success of the operation.
   * @throws {HttpException} Throws a FORBIDDEN exception if the authenticated user is not the owner of the store.
   */
  @Patch(':id')
  @ApiCookieAuth()
  @ApiParam({name: 'id', required: true, type: 'string' })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer)
  async update(@CurrentUser() user: ICurrentUser, @Param('id') id: Types.ObjectId, @Body() updateStoreDto: UpdateStoreDto): Promise<{ success: boolean; }> {
    // check same user tried to update
    if (String(user._id) !== String(updateStoreDto.user)) {
      throw new HttpException('Invalid Store', HttpStatus.FORBIDDEN);
    }
    await this.storeService.update(id, updateStoreDto);
    return { success: true };
  }

  /**
   * Removes a store from the system.
   * 
   * This endpoint is protected by authentication and role-based guards.
   * It deletes the store with the specified ID from the database.
   *
   * @param {Types.ObjectId} id - The unique identifier of the store to be removed.
   * @returns {Promise<any>} A promise that resolves to the result of the removal operation.
   *                         The exact return value depends on the implementation of storeService.remove().
   */
  @Delete(':id')
  @ApiCookieAuth()
  @ApiParam({name: 'id', required: true, type: 'string' })
  @UseGuards(AuthGuard, RoleGuard)
  async remove(@Param('id') id: Types.ObjectId): Promise<any> {
    return await this.storeService.remove(id);
  }
}
