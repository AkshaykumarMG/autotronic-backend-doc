import { Controller, Get, Post, Body, Patch, Param, Delete, UseGuards } from '@nestjs/common';
import { FilterService } from './filter.service';
import { CreateFilterDto } from './dto/create-filter.dto';
import { UpdateFilterDto } from './dto/update-filter.dto';
import { UpdateFilterSquenceDto } from './dto/update-filter-squence';

// mongoose
import { Types } from 'mongoose';
import { Filter } from './schema/filter.schema';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';

// swagger cookie
import { ApiBody, ApiCookieAuth, ApiParam } from '@nestjs/swagger';

@Controller('filter')
export class FilterController {
  constructor(private readonly filterService: FilterService) { }

  /**
   * Creates a new filter.
   * 
   * This endpoint is protected and can only be accessed by users with SuperAdmin role.
   * It requires authentication via cookie.
   * 
   * @param {CreateFilterDto} createFilterDto - The data transfer object containing the filter details to be created.
   * @returns {Promise<Filter>} The newly created filter object.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   */
  @Post()
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async create(@Body() createFilterDto: CreateFilterDto): Promise<Filter> {
    return await this.filterService.create(createFilterDto);
  }

  /**
   * Retrieves all filters.
   * 
   * This endpoint fetches all available filters from the database.
   * It's an unprotected route that can be accessed without authentication.
   * 
   * @returns {Promise<{success: boolean, data: Filter[]}>} An object containing:
   *   - success: A boolean indicating whether the operation was successful (always true in this case).
   *   - data: An array of Filter objects representing all the filters in the database.
   */
  @Get()
  async findAll(): Promise<{ success: boolean; data: Filter[]; }> {
    const result = await this.filterService.findAll();
    return { success: true, data: result }
  }

  /**
   * Retrieves all filters for admin users.
   * 
   * This endpoint is protected and can only be accessed by users with SuperAdmin or Admin roles.
   * It requires authentication via cookie.
   * 
   * @returns {Promise<{success: boolean, data: any}>} An object containing:
   *   - success: A boolean indicating whether the operation was successful (always true in this case).
   *   - data: The result returned by the filterService.findAllForAdmin() method, typically containing filter information for admin users.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   */
  @Get('admin')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin, UserRole.Admin)
  async findAllForAdmin(): Promise<{ success: boolean; data: any; }> {
    const result = await this.filterService.findAllForAdmin();
    return { success: true, data: result };
  }

  /**
   * Resets the count for all filters.
   * 
   * This endpoint is protected and can only be accessed by users with SuperAdmin role.
   * It requires authentication via cookie.
   * 
   * @returns {Promise<{ success: boolean }>} An object indicating the success status of the operation.
   *   - success: A boolean value that is true if the filter count reset was successful.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   */
  @Get('reset/count')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async resetFilterCount(): Promise<{ success: boolean; }> {
    await this.filterService.updateFilterCountReset();
    return { success: true };
  }

  /**
   * Retrieves a single filter by its ID.
   * 
   * This endpoint is protected and can only be accessed by users with SuperAdmin role.
   * It requires authentication via cookie.
   * 
   * @param {Types.ObjectId} id - The unique identifier of the filter to retrieve.
   * @returns {Promise<Filter>} A promise that resolves to the Filter object matching the provided ID.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   * @throws {NotFoundException} If no filter is found with the given ID.
   */
  @Get(':id')
  @ApiCookieAuth()
  @ApiParam({name: 'id', required: true, type: 'string' })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async findOne(@Param('id') id: Types.ObjectId): Promise<Filter> {
    return await this.filterService.findOne(id);
  }

  /**
   * Updates the sequence of filters.
   * 
   * This endpoint is protected and can only be accessed by users with SuperAdmin role.
   * It requires authentication via cookie.
   * 
   * @param {UpdateFilterSquenceDto[]} updateFilterSequenceDto - An array of objects containing filter IDs and their new sequence numbers.
   * @returns {Promise<{ success: boolean }>} An object indicating the success status of the operation.
   *   - success: A boolean value that is true if the filter sequence update was successful.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   */
  @Patch('sequence')
  @ApiCookieAuth()
  @ApiBody({ type: [UpdateFilterSquenceDto] })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async updateFilterSequence(@Body() updateFilterSequenceDto: UpdateFilterSquenceDto[]): Promise<{ success: boolean }> {
    await this.filterService.updateFilterSequence(updateFilterSequenceDto);
    return { success: true }
  }

  /**
   * Updates a specific filter by its ID.
   * 
   * This endpoint is protected and can only be accessed by users with SuperAdmin role.
   * It requires authentication via cookie.
   * 
   * @param {Types.ObjectId} id - The unique identifier of the filter to update.
   * @param {UpdateFilterDto} updateFilterDto - The data transfer object containing the updated filter details.
   * @returns {Promise<Filter>} A promise that resolves to the updated Filter object.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   * @throws {NotFoundException} If no filter is found with the given ID.
   */
  @Patch(':id')
  @ApiCookieAuth()
  @ApiParam({name: 'id', required: true, type: 'string' })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async update(@Param('id') id: Types.ObjectId, @Body() updateFilterDto: UpdateFilterDto) {
    return await this.filterService.update(id, updateFilterDto);
  }

  /**
   * Deletes a specific filter by its ID.
   * 
   * This endpoint is protected and can only be accessed by users with SuperAdmin role.
   * It requires authentication via cookie.
   * 
   * @param {Types.ObjectId} id - The unique identifier of the filter to be deleted.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success status of the operation.
   *   - success: A boolean value that is true if the filter was successfully deleted.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   * @throws {NotFoundException} If no filter is found with the given ID.
   */
  @Delete(':id')
  @ApiCookieAuth()
  @ApiParam({name: 'id', required: true, type: 'string' })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async remove(@Param('id') id: Types.ObjectId): Promise<{ success: boolean }> {
    await this.filterService.remove(id);
    return { success: true };
  }
}
