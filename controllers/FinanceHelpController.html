import { Controller, Get, Post, Body, Patch, Param, Delete, Query, UseGuards } from '@nestjs/common';
import { FinanceHelpService } from './finance-help.service';
import { CreateFinanceHelpDto } from './dto/create-finance-help.dto';
// import { UpdateFinanceHelpDto } from './dto/update-finance-help.dto';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';
import { ManageFinanceHelpDto } from './dto/manage-finance-help.dto';

// swagger cookie
import { ApiCookieAuth, ApiParam } from '@nestjs/swagger';

// get status of finance help
import { FinanceHelpStatus } from './schema/finance-help.schema';

@Controller('finance-help')
export class FinanceHelpController {
  constructor(private readonly financeHelpService: FinanceHelpService) { }
  /**
   * Creates a new finance help request.
   * 
   * This method handles the creation of a new finance help request by calling the
   * financeHelpService's create method with the provided data.
   *
   * @param {CreateFinanceHelpDto} createFinanceHelpDto - The DTO containing the data for creating a new finance help request.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success status of the operation.
   */
  @Post()
  async create(@Body() createFinanceHelpDto: CreateFinanceHelpDto): Promise<{ success: boolean; }> {
    await this.financeHelpService.create(createFinanceHelpDto);
    return { success: true };
  }

  /**
   * Retrieves the count of finance help requests for each status.
   * 
   * This endpoint is protected and can only be accessed by users with Admin or SuperAdmin roles.
   * It returns a count of finance help requests grouped by their status.
   *
   * @returns {Promise<object>} A promise that resolves to an object containing the count of finance help requests for each status.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   */
  @Get('status')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Admin, UserRole.SuperAdmin)
  async findAllStoreStatusWithCount(): Promise<object> {
    return await this.financeHelpService.getFinanceHelpStatusWithCount();
  }

  /**
   * Retrieves finance help requests for admin users based on specified status and pagination parameters.
   * 
   * This endpoint is protected and can only be accessed by users with Admin or SuperAdmin roles.
   * It returns a paginated list of finance help requests filtered by the given status.
   *
   * @param {string} status - The status of finance help requests to filter by. Must be a valid FinanceHelpStatus enum value.
   * @param {ICurrentUser} user - The current authenticated user object.
   * @param {number} page - The page number for pagination.
   * @param {number} limit - The number of items per page for pagination.
   * @returns {Promise<object>} A promise that resolves to an object containing the paginated list of finance help requests.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   */
  @Get('admin/:status')
  @ApiCookieAuth()
  @ApiParam({ name: 'status', required: true, enum: FinanceHelpStatus })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Admin, UserRole.SuperAdmin)
  async findAllStoreForAdmin(@Param('status') status: string, @CurrentUser() user: ICurrentUser, @Query('page') page: number, @Query('limit') limit: number): Promise<object> {
    return await this.financeHelpService.findAllForAdmin(status, user, page, limit);
  }

  // @Get(':id')

  /**
   * Updates the status of a finance help request.
   * 
   * This endpoint is protected and can only be accessed by users with Admin or SuperAdmin roles.
   * It allows authorized users to update the status of a finance help request.
   *
   * @param {ICurrentUser} user - The current authenticated user object.
   * @param {ManageFinanceHelpDto} manageFinanceHelpDto - The DTO containing the data for updating the finance help status.
   * @returns {Promise<any>} A promise that resolves to the result of the update operation.
   * @throws {UnauthorizedException} If the user is not authenticated or doesn't have the required role.
   */
  @Patch('status')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Admin, UserRole.SuperAdmin)
  async updateFinanceStatus(@CurrentUser() user: ICurrentUser, @Body() manageFinanceHelpDto: ManageFinanceHelpDto): Promise<any> {
    return await this.financeHelpService.updateFinanceStatus(user, manageFinanceHelpDto);
  }
