import { Controller, Get, Post, Body, Patch, Param, Delete, UseGuards } from '@nestjs/common';
import { FinanceApplicationService } from './finance-application.service';
import { CreateFinanceApplicationDto } from './dto/create-finance-application.dto';
import { UpdateFinanceApplicationDto } from './dto/update-finance-application.dto';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';
import { Types } from 'mongoose';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';

// swagger cookie
import { ApiBody, ApiCookieAuth, ApiParam } from '@nestjs/swagger';
import { data } from '../car/trim.data';

@Controller('finance-application')
@ApiCookieAuth()
@UseGuards(AuthGuard, RoleGuard)
export class FinanceApplicationController {
  constructor(private readonly financeApplicationService: FinanceApplicationService) { }

  /**
   * Creates a new finance application.
   * 
   * This endpoint is accessible to users with Dealer or Person roles.
   * It creates a new finance application based on the provided data.
   *
   * @param {CreateFinanceApplicationDto} createFinanceApplicationDto - The DTO containing the finance application data.
   * @returns {Promise<any>} The created finance application object.
   */
  @Post()
  @Roles(UserRole.Dealer, UserRole.Person)
  async create(@Body() createFinanceApplicationDto: CreateFinanceApplicationDto): Promise<any> {
    return await this.financeApplicationService.create(createFinanceApplicationDto);
  }

  /**
   * Retrieves the finance application for the current user.
   * 
   * This endpoint fetches the finance application associated with the authenticated user.
   * It uses the CurrentUser decorator to access the user's information from the request.
   *
   * @param {ICurrentUser} user - The current authenticated user, injected by the CurrentUser decorator.
   * @returns {Promise<any>} A promise that resolves to the finance application data for the user.
   *                         The exact structure of the returned data depends on the implementation
   *                         of the financeApplicationService.findOne method.
   */
  @Get()
  async findOne(@CurrentUser() user: ICurrentUser): Promise<any> {
    return await this.financeApplicationService.findOne(user);
  }

  /**
   * Retrieves the finance application for the authenticated user.
   * 
   * This endpoint is accessible to users with Dealer or Person roles.
   * It fetches the finance application associated with the current user's ID.
   *
   * @param {ICurrentUser} user - The current authenticated user, injected by the CurrentUser decorator.
   *                              Contains user information including the user's ID.
   * 
   * @returns {Promise<{success: boolean, data?: any}>} A promise that resolves to an object containing:
   *   - success: A boolean indicating whether the operation was successful.
   *   - data: The finance application data if found. This property is omitted if no data is found.
   */
  @Get('user')
  @Roles(UserRole.Dealer, UserRole.Person)
  async findOneByUser(@CurrentUser() user: ICurrentUser): Promise<{ success: boolean; data?: any; }> {
    const result =  await this.financeApplicationService.findOneByUserId(user._id);
    if (!result) {
      return { success: false };
    }
    return { success: true, data: result };
  }

  /**
   * Updates an existing finance application.
   * 
   * This endpoint replaces the entire finance application with the provided data.
   * It requires the full create object, not just updates to specific fields.
   *
   * @param {Types.ObjectId} id - The unique identifier of the finance application to update.
   * @param {UpdateFinanceApplicationDto} updateFinanceApplicationDto - The DTO containing the complete updated finance application data.
   * @returns {Promise<any>} A promise that resolves to the updated finance application object.
   */
  @Patch(':id')
  @ApiBody({ description: 'Write whole create object, not just update the primary applicant fields, this API is replace whole update.' })
  @ApiParam({name: 'id', required: true, type: 'string' })
  async update(@Param('id') id: Types.ObjectId, @Body() updateFinanceApplicationDto: UpdateFinanceApplicationDto): Promise<any> {
    return await this.financeApplicationService.update(id, updateFinanceApplicationDto);
  }

  /**
   * Removes a finance application.
   * 
   * This endpoint deletes a specific finance application based on the provided ID.
   * It is decorated with @Delete and @ApiParam for Swagger documentation.
   *
   * @param {Types.ObjectId} id - The unique identifier of the finance application to be removed.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success of the operation.
   *                                         Returns { success: true } if the removal was successful.
   */
  @Delete(':id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  async remove(@Param('id') id: Types.ObjectId): Promise<{ success: boolean }> {
    await this.financeApplicationService.remove(id);
    return { success: true };
  }
}
