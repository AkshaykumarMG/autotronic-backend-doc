import { Controller, Get, Post, Body, Patch, Param, Delete, UseGuards, Query } from '@nestjs/common';
import { AwardService } from './award.service';
import { CreateAwardDto } from './dto/create-award.dto';
import { UpdateAwardDto } from './dto/update-award.dto';
import { ManageUserAwardDto } from './dto/manage-user-award.dto';

// mongoose
import { Types } from 'mongoose';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';

// swagger cookie
import { ApiCookieAuth, ApiParam } from '@nestjs/swagger';
import { Award } from './schema/award.schema';

@Controller('award')
@ApiCookieAuth()
@UseGuards(AuthGuard, RoleGuard)
@Roles(UserRole.SuperAdmin)
export class AwardController {
  constructor(private readonly awardService: AwardService) { }

  /**
   * Creates a new award in the system.
   * 
   * This function handles the creation of a new award by setting the creator
   * and passing the data to the award service.
   * 
   * @param user - The current authenticated user creating the award.
   * @param createAwardDto - The data transfer object containing the award details.
   * @returns A promise that resolves to the created award object.
   */
  @Post()
  async create(@CurrentUser() user: ICurrentUser, @Body() createAwardDto: CreateAwardDto) {
    createAwardDto.createdBy = user._id;
    return await this.awardService.create(createAwardDto);
  }

  /**
   * Adds an award to a store.
   * 
   * This function handles the process of associating an award with a specific store.
   * It calls the award service to update the store's award list.
   * 
   * @param manageUserAwardDto - Data transfer object containing the award and store information.
   *                             This includes details such as the award ID and the store ID.
   * @returns A promise that resolves to an object indicating the success of the operation.
   *          The object has a single boolean property 'success' which is true if the
   *          award was successfully added to the store.
   */
  @Post('add-store')
  async addAwardToStore(@Body() manageUserAwardDto: ManageUserAwardDto) {
    await this.awardService.updateAddAwardToStore(manageUserAwardDto);
    return { success: true };
  }

  /**
   * Removes an award from a store.
   * 
   * This function handles the process of disassociating an award from a specific store.
   * It calls the award service to update the store's award list by removing the specified award.
   * 
   * @param {ManageUserAwardDto} manageUserAwardDto - Data transfer object containing the award and store information.
   *                                                  This includes details such as the award ID and the store ID.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success of the operation.
   *                                          The object has a single boolean property 'success' which is true if the
   *                                          award was successfully removed from the store.
   */
  @Post('remove-store')
  async removeAwardToStore(@Body() manageUserAwardDto: ManageUserAwardDto): Promise<{ success: boolean; }> {
    await this.awardService.updateRmoveAwardToStore(manageUserAwardDto);
    return { success: true };
  }

  /**
   * Retrieves a paginated list of all awards.
   * 
   * This endpoint allows fetching awards with pagination support. It uses query parameters
   * to determine which page of results to return and how many items per page.
   *
   * @param {number} page - The page number of results to return. Used for pagination.
   * @param {number} limit - The maximum number of items to return per page.
   * @returns {Promise<any>} A promise that resolves to an object containing the paginated list of awards.
   *                         The exact structure of the return object depends on the implementation
   *                         of the AwardService's findAll method.
   */
  @Get()
  async findAll(@Query('page') page: number, @Query('limit') limit: number): Promise<any> {
    return await this.awardService.findAll(page, limit);
  }

  /**
   * Retrieves all awards for selection purposes.
   * 
   * This endpoint fetches a list of all awards in a format suitable for selection,
   * typically used in dropdown menus or similar UI components. The exact format
   * of the returned data depends on the implementation of the AwardService's
   * findAllForSelect method.
   *
   * @returns {Promise<any>} A promise that resolves to an array or object containing
   *                         the awards in a format suitable for selection. The exact
   *                         structure depends on the AwardService implementation.
   */
  @Get('select')
  async findAllForSelect(): Promise<any> {
    return await this.awardService.findAllForSelect();
  }

  /**
   * Retrieves a specific award by its ID.
   * 
   * This endpoint fetches the details of a single award based on the provided ID.
   * It uses the AwardService to fetch the award from the database.
   *
   * @param {Types.ObjectId} id - The unique identifier of the award to retrieve.
   *                              This should be a valid MongoDB ObjectId.
   * @returns {Promise<Award>} A promise that resolves to the Award object matching the provided ID.
   *                           The exact structure of the Award object depends on the schema definition.
   * @throws {NotFoundException} If no award is found with the given ID.
   */
  @Get(':id')
  @ApiParam({ name: 'id', required: true, type: 'string' })
  async findOne(@Param('id') id: Types.ObjectId): Promise<Award> {
    return await this.awardService.findOne(id);
  }

  /**
   * Updates an existing award in the system.
   * 
   * This endpoint allows for the modification of an award's details based on the provided ID.
   * It uses the AwardService to update the award in the database.
   *
   * @param {Types.ObjectId} id - The unique identifier of the award to be updated.
   *                              This should be a valid MongoDB ObjectId.
   * @param {UpdateAwardDto} updateAwardDto - Data transfer object containing the updated award details.
   *                                          This includes any fields that need to be modified.
   * @returns {Promise<Award>} A promise that resolves to the updated Award object.
   *                           The returned object reflects the award's state after the update.
   * @throws {NotFoundException} If no award is found with the given ID.
   */
  @Patch(':id')
  @ApiParam({ name: 'id', required: true, type: 'string' })
  async update(@Param('id') id: Types.ObjectId, @Body() updateAwardDto: UpdateAwardDto): Promise<Award> {
    return await this.awardService.update(id, updateAwardDto);
  }

  /**
   * Removes an award from the system.
   * 
   * This endpoint deletes a specific award based on the provided ID.
   * It uses the AwardService to remove the award from the database.
   *
   * @param {Types.ObjectId} id - The unique identifier of the award to be removed.
   *                              This should be a valid MongoDB ObjectId.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object 
   *                                          indicating the success of the operation.
   *                                          The object has a single boolean property 
   *                                          'success' which is true if the award 
   *                                          was successfully removed.
   * @throws {NotFoundException} If no award is found with the given ID.
   */
  @Delete(':id')
  @ApiParam({ name: 'id', required: true, type: 'string' })
  async remove(@Param('id') id: Types.ObjectId): Promise<{ success: boolean }> {
    await this.awardService.remove(id);
    return { success: true };
  }
}
