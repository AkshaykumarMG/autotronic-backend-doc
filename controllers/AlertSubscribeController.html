import { Controller, Get, Post, Body, Param, Delete, Req, UseGuards } from '@nestjs/common';
import { AlertSubscribeService } from './alert-subscribe.service';
import { CreateAlertSubscribeDto } from './dto/create-alert-subscribe.dto';
// import { UpdateAlertSubscribeDto } from './dto/update-alert-subscribe.dto';
import { Request } from 'express';
import { Types } from 'mongoose';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';

// swagger cookie
import { ApiCookieAuth, ApiParam } from '@nestjs/swagger';

@Controller('alert-subscribe')
export class AlertSubscribeController {
  constructor(private readonly alertSubscribeService: AlertSubscribeService) {}

  /**
   * Creates a new alert subscription for the user.
   * This endpoint allows users to subscribe to alerts based on the provided data.
   *
   * @param {Request} request - The Express request object containing information about the HTTP request.
   * @param {CreateAlertSubscribeDto} createAlertSubscribeDto - The DTO (Data Transfer Object) containing the data for creating a new alert subscription.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success of the operation.
   *                                         Returns { success: true } if the alert subscription is created successfully.
   * @throws {Error} If there's an issue creating the alert subscription.
   */
  @Post()
  async create(@Req() request: Request, @Body() createAlertSubscribeDto: CreateAlertSubscribeDto): Promise<{ success: boolean; }> {
    await this.alertSubscribeService.create(request, createAlertSubscribeDto);
    return { success: true };
  }

  /**
   * Retrieves all alert subscriptions for the authenticated user.
   * This endpoint is protected and requires authentication and authorization.
   * Only users with Dealer or Person roles can access this endpoint.
   *
   * @param {ICurrentUser} user - The current authenticated user, obtained from the @CurrentUser() decorator.
   * @returns {Promise<any[]>} A promise that resolves to an array of alert subscriptions for the user.
   *                           If no subscriptions are found, an empty array is returned.
   * @throws {UnauthorizedException} If the user is not authenticated or lacks the required role.
   */
  @Get('user')
  @ApiCookieAuth()
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer, UserRole.Person)
  async findOne(@CurrentUser() user: ICurrentUser) {
    const data = await this.alertSubscribeService.findAllByUserEmail(user.email);
    return data || [];
  }

  /**
   * Removes an alert subscription by its ID.
   * This endpoint is protected and requires authentication and authorization.
   * Only users with Dealer or Person roles can access this endpoint.
   *
   * @param {Types.ObjectId} id - The unique identifier of the alert subscription to be removed.
   * @returns {Promise<{ success: boolean }>} A promise that resolves to an object indicating the success of the operation.
   * @throws {UnauthorizedException} If the user is not authenticated or lacks the required role.
   * @throws {NotFoundException} If the alert subscription with the given ID is not found.
   */
  @Delete(':id')
  @ApiCookieAuth()
  @ApiParam({name: 'id', required: true, type: 'string' })
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Dealer, UserRole.Person)
  async remove(@Param('id') id: Types.ObjectId) {
    await this.alertSubscribeService.remove(id);
    return { success: true };
  }
}
