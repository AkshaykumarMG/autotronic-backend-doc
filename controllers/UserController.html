import { Controller, Get, Post, Body, Patch, Param, Delete, UseGuards, Query, Inject } from '@nestjs/common';
import { UserService } from './user.service';
// import { CreateUserDto } from './dto/create-user.dto';
import { UpdateUserDto } from './dto/update-user.dto';
import { Types } from 'mongoose';
import { CreateUserAdminDto } from './dto/create-user-admin.dto';

// custom service
import { RedisService } from '../../utilities/services/redis/redis.service';

// verify user by credential and role
import { AuthGuard } from '../../utilities/guard/auth/auth.guard';
import { Roles } from '../user/role/role.decorator';
import { UserRole } from '../user/schema/user.schema';
import { RoleGuard } from '../../utilities/guard/role/role.guard';
import { CurrentUser, ICurrentUser } from '../user/decorator/current-user.decorator';

// custom module type cast
import { ChangePasswordDto } from './dto/change-password.dto';
import { SetUserAdminPermissionDto } from './dto/set-user-admin-permission.dto';
import { SetStorePermissionDto } from './dto/set-store-permission.dto';
import { CreateAddressDTO } from './dto/create-address.dto';
import { ManageUserFavouriteDto } from './dto/manage-user-favourite.dto';

// swagger cookie
import { ApiCookieAuth, ApiParam } from '@nestjs/swagger';

@Controller('user')
@ApiCookieAuth()
@UseGuards(AuthGuard, RoleGuard)
export class UserController {
  constructor(
    private readonly userService: UserService,
    @Inject(RedisService) private readonly redisService: RedisService,
  ) { }

  @Post()
  @Roles(UserRole.SuperAdmin)
  create(@Body() createUserDto: CreateUserAdminDto) {
    return this.userService.create(createUserDto);
  }

  @Post('admin/store')
  @Roles(UserRole.SuperAdmin)
  async setUserAdminPermission(@CurrentUser() user: ICurrentUser, @Body() setUserAdminPermissionDto: SetUserAdminPermissionDto) {
    await this.userService.updateUserAdminStore(user._id, setUserAdminPermissionDto);
    return { success: true };
  }

  @Post('store/status')
  @Roles(UserRole.SuperAdmin)
  async setStoreByUser(@CurrentUser() user: ICurrentUser, @Body() setStorePermissionDto: SetStorePermissionDto) {
    await this.userService.updateUserStoreOperation(user._id, setStorePermissionDto);
    return { success: true };
  }

  @Post('change-password')
  async changePassword(@CurrentUser() user: ICurrentUser, @Body() changePasswordDto: ChangePasswordDto) {
    await this.userService.changePassword(user._id, changePasswordDto);
    return { success: true };
  }

  @Get()
  @Roles(UserRole.SuperAdmin)
  async findAll(@Query('page') page: number, @Query('limit') limit: number) {
    return await this.userService.findAll(page, limit);
  }


  /* This code block in the `UserController` class is defining a GET endpoint `/user/count/total` that
  is used to retrieve the total count of users. Here is a breakdown of what each part of the code is
  doing: */
  @Get('count/total')
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.SuperAdmin)
  async findAllCarForCount() {
    return await this.redisService.get('all:user');
  }

  /* This specific code block in the UserController class is defining a GET endpoint
  '/user/count/store/total' that is used to retrieve the total count of stores associated with a
  specific user who has the roles of 'SuperAdmin' or 'Admin'. Here is a breakdown of what each part
  of the code is doing: */
  @Get('count/store/total')
  @UseGuards(AuthGuard, RoleGuard)
  @Roles(UserRole.Admin)
  async findAllCarForCountForAdmin(@CurrentUser() user: ICurrentUser) {
    const result = await this.userService.findByIDWithStores(user._id);
    return result ? result.store.length : 0;
  }

  @Get('select')
  @Roles(UserRole.SuperAdmin)
  async findAllUserWithIDandEmail() {
    return await this.userService.findAllByIDandEmail();
  }

  @Get('status/admin')
  @Roles(UserRole.SuperAdmin)
  async findAllUserWithAdminStatus() {
    return await this.userService.findAllByAdminStatus();
  }

  @Get('store')
  async findByCurrentUserForStore(@CurrentUser() user: ICurrentUser) {
    return await this.userService.findByIDWithStores(user._id);
  }

  @Get('profile')
  async findByCurrentUserForprofile(@CurrentUser() user: ICurrentUser) {
    return await this.userService.findByID(user._id);
  }

  @Get('identity')
  async findByCurrentUser(@CurrentUser() user: ICurrentUser) {
    return await this.userService.findByIDWithID(user._id);
  }

  /**
 * Retrieves the list of favorite vehicles for the currently authenticated user.
 *
 * @remarks
 * This function is used to fetch the list of favorite vehicles for the user identified by the
 * `user.email` property of the `ICurrentUser` object, which is obtained from the `@CurrentUser()` decorator.
 *
 * @param user - The current authenticated user.
 * @returns A Promise that resolves to an array of favorite vehicle IDs.
 *
 * @example
 * ```typescript
 * const user: ICurrentUser = { _id: '123', email: 'user@example.com', role: UserRole.Person };
 * const favoriteVehicles = await findUserFavouriteVahicles(user);
 * console.log(favoriteVehicles); // Output: ['vehicle1', 'vehicle2', ...]
 * ```
 */
  @Get('favourite-list')
  async findUserFavouriteVahicles(@CurrentUser() user: ICurrentUser) {
    return await this.userService.findUserFavouriteVehicles(user.email);
  }

  @Get(':id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  @Roles(UserRole.SuperAdmin)
  async findOne(@Param('id') id: Types.ObjectId) {
    return await this.userService.findOne(id);
  }

  @Get('status/:id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  @Roles(UserRole.SuperAdmin)
  async findUsersByStoreID(@Param('id') id: Types.ObjectId) {
    return await this.userService.findUsersByStoreID(id);
  }

  @Get('store/:id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  async findOneUserForStore(@Param('id') id: Types.ObjectId) {
    return await this.userService.findOneWithStore(id);
  }

  @Patch('favourite')
  @Roles(UserRole.Person, UserRole.Dealer)
  async manageUserFavourite(@CurrentUser() user: ICurrentUser, @Body() manageUserFavouriteDto: ManageUserFavouriteDto) {
    await this.userService.manageFavoriteUserAdvertisement(user._id, manageUserFavouriteDto);
    return { success: true };
  }
  
  @Patch(':id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  @Roles(UserRole.SuperAdmin)
  async update(@Param('id') id: Types.ObjectId, @Body() updateUserDto: UpdateUserDto) {
    return await this.userService.update(id, updateUserDto);
  }

  /**
 * Updates the address of a user by their ID.
 */
  @Patch('address/:id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  @Roles(UserRole.Person)
  async updateAddress(@Param('id') id: Types.ObjectId, @Body() createAddressDTO: CreateAddressDTO) {
    return await this.userService.updateUserAddress(id, createAddressDTO);
  }

  @Delete(':id')
  @ApiParam({name: 'id', required: true, type: 'string' })
  @Roles(UserRole.SuperAdmin)
  async remove(@Param('id') id: Types.ObjectId) {
    await this.userService.remove(id);
    return { success: true };
  }
}
