Description

Defines the Filter module configuration for the application. This module sets up the necessary imports, controllers, providers, and exports for handling filter-related functionality.

  • MongooseModule.forFeature: Sets up the Filter schema for MongoDB.
  • CarModule (forward reference): Imports the Car module to handle car-related operations.
  • UserModule: Imports the User module for user-related functionality.
  • AuthenticationModule: Imports the Authentication module for authentication services.
  • FilterController: Handles HTTP requests related to filters.
  • FilterService: Provides filter-related business logic.
  • RedisService: Handles Redis caching operations.
  • FilterService: Makes the FilterService available for use in other modules.

import { forwardRef, Module } from '@nestjs/common';
import { FilterService } from './filter.service';
import { FilterController } from './filter.controller';

// mongodb - filter model implementation
import { MongooseModule } from '@nestjs/mongoose';
import { Filter, FilterSchema } from './schema/filter.schema';

// redis
import { RedisService } from '../../utilities/services/redis/redis.service';

// add car schema to add all selected filter's in filter schema
import { CarModule } from '../car/car.module';

// user permission & guard
import { UserModule } from '../user/user.module';
import { AuthenticationModule } from '../authentication/authentication.module';

/**
 * Defines the Filter module configuration for the application.
 * This module sets up the necessary imports, controllers, providers, and exports
 * for handling filter-related functionality.
 *
 * @module
 * @param {Object} options - The module configuration options.
 * @param {Array} options.imports - An array of imported modules:
 *   - MongooseModule.forFeature: Sets up the Filter schema for MongoDB.
 *   - CarModule (forward reference): Imports the Car module to handle car-related operations.
 *   - UserModule: Imports the User module for user-related functionality.
 *   - AuthenticationModule: Imports the Authentication module for authentication services.
 * @param {Array} options.controllers - An array of controllers used in this module:
 *   - FilterController: Handles HTTP requests related to filters.
 * @param {Array} options.providers - An array of providers (services) used in this module:
 *   - FilterService: Provides filter-related business logic.
 *   - RedisService: Handles Redis caching operations.
 * @param {Array} options.exports - An array of providers exported for use in other modules:
 *   - FilterService: Makes the FilterService available for use in other modules.
 * @returns {Object} The configured Filter module.
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: Filter.name, schema: FilterSchema }]),
    forwardRef(() => CarModule), UserModule, AuthenticationModule
  ],
  controllers: [FilterController],
  providers: [FilterService, RedisService],
  exports: [FilterService]
})
export class FilterModule { }
