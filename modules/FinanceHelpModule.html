This module integrates various components and services related to finance help, including database models, authentication, notifications, and vehicle finance applications.

import { Module } from '@nestjs/common';
import { FinanceHelpService } from './finance-help.service';
import { FinanceHelpController } from './finance-help.controller';

// mongodb
import { MongooseModule } from '@nestjs/mongoose';
import { FinanceHelp, FinanceHelpSchema } from './schema/finance-help.schema';

// permission module's
import { UserModule } from '../user/user.module';
import { AuthenticationModule } from '../authentication/authentication.module';

// mail service
import { MailService } from '../../utilities/services/mail/mail.service';

// notification
import { NotificationModule } from '../notification/notification.module';
import { FirebaseNotificationService } from '../../utilities/services/firebase/notification/firebase-notification.service';

// sub relationship
import { VehicleFinanceApplicationModule } from '../vehicle-finance-application/vehicle-finance-application.module';

/**
 * Defines the FinanceHelpModule, which encapsulates the finance help functionality of the application.
 * 
 * This module integrates various components and services related to finance help, including
 * database models, authentication, notifications, and vehicle finance applications.
 * 
 * @module
 * @param {Object} options - The module configuration options.
 * @param {Array} options.imports - An array of imported modules required for this module.
 * @param {Array} options.controllers - An array of controllers used in this module.
 * @param {Array} options.providers - An array of providers (services) used in this module.
 * @returns {Module} A NestJS module that can be used in the application.
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: FinanceHelp.name, schema: FinanceHelpSchema }]),
    UserModule,
    AuthenticationModule,
    NotificationModule,
    VehicleFinanceApplicationModule
  ],
  controllers: [FinanceHelpController],
  providers: [FinanceHelpService, MailService, FirebaseNotificationService],
})
export class FinanceHelpModule { }
