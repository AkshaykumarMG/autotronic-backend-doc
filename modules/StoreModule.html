Description

Defines the Store module for the application. This module handles store-related functionality, including database operations, authentication, and notifications.

  • MongooseModule.forFeature: Registers the Store schema with Mongoose.
  • UserModule: Imported using forwardRef to avoid circular dependencies.
  • AuthenticationModule: Handles authentication-related functionality.
  • NotificationModule: Manages notification services.
  • StoreController: Handles HTTP requests related to stores.
  • StoreService: Provides store-related business logic.
  • AwsS3Service: Handles AWS S3 operations for file storage.
  • MailService: Manages email-related functionality.
  • FirebaseNotificationService: Handles Firebase push notifications.
  • StoreService: Makes the StoreService available for use in other modules.

import { Module, forwardRef } from '@nestjs/common';
import { StoreService } from './store.service';
import { StoreController } from './store.controller';

// mongoose
import { MongooseModule } from '@nestjs/mongoose';
import { Store, StoreSchema } from './schema/store.schema';

// permission & auth
import { UserModule } from '../user/user.module';
import { AuthenticationModule } from '../authentication/authentication.module';

// media service
import { AwsS3Service } from '../../utilities/services/aws_s3/aws_s3.service';

// mail serice
import { MailService } from '../../utilities/services/mail/mail.service';

// notification service
import { FirebaseNotificationService } from '../../utilities/services/firebase/notification/firebase-notification.service';
import { NotificationModule } from '../notification/notification.module';

/**
 * Defines the Store module for the application.
 * This module handles store-related functionality, including database operations,
 * authentication, and notifications.
 * 
 * @decorator @Module
 * @param {Object} options - The module configuration options.
 * @param {Array} options.imports - The list of imported modules.
 *   - MongooseModule.forFeature: Registers the Store schema with Mongoose.
 *   - UserModule: Imported using forwardRef to avoid circular dependencies.
 *   - AuthenticationModule: Handles authentication-related functionality.
 *   - NotificationModule: Manages notification services.
 * @param {Array} options.controllers - The controllers to be instantiated within the module.
 *   - StoreController: Handles HTTP requests related to stores.
 * @param {Array} options.providers - The providers to be instantiated within the module.
 *   - StoreService: Provides store-related business logic.
 *   - AwsS3Service: Handles AWS S3 operations for file storage.
 *   - MailService: Manages email-related functionality.
 *   - FirebaseNotificationService: Handles Firebase push notifications.
 * @param {Array} options.exports - The providers to be exported and available for import in other modules.
 *   - StoreService: Makes the StoreService available for use in other modules.
 * @returns {void}
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: Store.name, schema: StoreSchema }]),
    forwardRef(() => UserModule),
    AuthenticationModule,
    NotificationModule
  ],
  controllers: [StoreController],
  providers: [StoreService, AwsS3Service, MailService, FirebaseNotificationService],
  exports: [StoreService]
})
export class StoreModule {}
