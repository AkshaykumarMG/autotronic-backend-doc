Description

Defines the AlertPriceModule, which manages price alerts for cars.

This module integrates various components and services related to price alerts, including database models, authentication, user management, and email notifications.

The module uses MongoDB for data persistence and includes services for handling alert subscriptions and sending email notifications.

import { Module, forwardRef } from '@nestjs/common';
import { AlertPriceService } from './alert-price.service';
import { AlertPriceController } from './alert-price.controller';

// mongoose
import { MongooseModule } from '@nestjs/mongoose';
import { AlertPrice, AlertPriceSchema } from './schema/alert-price.schema';

// permission & authentication
import { UserModule } from '../user/user.module';
import { AuthenticationModule } from '../authentication/authentication.module';

// mail service
import { MailService } from '../../utilities/services/mail/mail.service';

// car module
import { CarModule } from '../car/car.module';

// alert subscribe consumer service
import { AlertSubscribeConsumerService } from './queue/alert-subscription-consumer.service';

/**
 * Defines the AlertPriceModule, which manages price alerts for cars.
 * 
 * This module integrates various components and services related to price alerts,
 * including database models, authentication, user management, and email notifications.
 * 
 * @remarks
 * The module uses MongoDB for data persistence and includes services for 
 * handling alert subscriptions and sending email notifications.
 * 
 * @module AlertPriceModule
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: AlertPrice.name, schema: AlertPriceSchema }]),
    UserModule,
    AuthenticationModule,
    forwardRef(() => CarModule)
  ],
  controllers: [AlertPriceController],
  providers: [AlertPriceService, MailService, AlertSubscribeConsumerService],
  exports: [AlertPriceService]
})
export class AlertPriceModule {}
