Description

Defines the Award module for the application. This module encapsulates all award-related functionality, including database interactions, controllers, services, and dependencies.

  • MongooseModule.forFeature: Registers the Award schema with Mongoose.
  • StoreModule: Provides store-related functionality.
  • UserModule: Handles user-related operations.
  • AuthenticationModule: Manages authentication processes.
  • AwardController: Handles HTTP requests related to awards.
  • AwardService: Provides award-related business logic.
  • AwsS3Service: Handles AWS S3 operations, likely for award image storage.

import { Module } from '@nestjs/common';
import { AwardService } from './award.service';
import { AwardController } from './award.controller';

// mongoose
import { Award, AwardSchema } from './schema/award.schema';
import { MongooseModule } from '@nestjs/mongoose';

// permission & authentication
import { UserModule } from '../user/user.module';
import { AuthenticationModule } from '../authentication/authentication.module';

// file image remove
import { AwsS3Service } from '../../utilities/services/aws_s3/aws_s3.service';

// store to manage relationship
import { StoreModule } from '../store/store.module';

/**
 * Defines the Award module for the application.
 * This module encapsulates all award-related functionality, including database interactions,
 * controllers, services, and dependencies.
 *
 * @decorator @Module
 * @param {Object} options - The module configuration options.
 * @param {Array} options.imports - An array of imported modules required for this module.
 *   - MongooseModule.forFeature: Registers the Award schema with Mongoose.
 *   - StoreModule: Provides store-related functionality.
 *   - UserModule: Handles user-related operations.
 *   - AuthenticationModule: Manages authentication processes.
 * @param {Array} options.controllers - An array of controllers to be instantiated within this module.
 *   - AwardController: Handles HTTP requests related to awards.
 * @param {Array} options.providers - An array of providers (services) to be instantiated within this module.
 *   - AwardService: Provides award-related business logic.
 *   - AwsS3Service: Handles AWS S3 operations, likely for award image storage.
 * @returns {Module} The configured Award module.
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: Award.name, schema: AwardSchema }]),
    StoreModule,
    UserModule,
    AuthenticationModule
  ],
  controllers: [AwardController],
  providers: [AwardService, AwsS3Service],
})
export class AwardModule {}
