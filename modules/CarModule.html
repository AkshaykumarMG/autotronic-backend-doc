Description

Defines the CarModule, which encapsulates all car-related functionality.

This module imports necessary dependencies, defines controllers and providers, and exports the CarService for use in other modules.

The module uses MongooseModule for database operations, and imports various related modules such as FilterModule, UserModule, AuthenticationModule, AlertPriceModule, and HttpModule.

import { Module, forwardRef } from '@nestjs/common';
import { CarService } from './car.service';
import { CarController } from './car.controller';

// mongodb
import { Car, CarSchema } from './schema/car.schema';
import { MongooseModule } from '@nestjs/mongoose';

// Add & update car, media upload by queue
// import { ClientsModule, Transport } from '@nestjs/microservices';
// import { config } from '../../utilities/config';
import { CarConsumerService } from './queue/car-consumer.service';

// add filter module - to update filter count's
import { FilterModule } from '../filter/filter.module';

// permission and guard
import { UserModule } from '../user/user.module';
import { AuthenticationModule } from '../authentication/authentication.module';
import { AlertPriceModule } from '../alert-price/alert-price.module';

// http module to get location information
import { HttpModule } from '@nestjs/axios';

// aws s3 service
import { AwsS3Service } from '../../utilities/services/aws_s3/aws_s3.service';

// mail service for sending emails
import { MailService } from '../../utilities/services/mail/mail.service';

// redis
import { RedisService } from '../../utilities/services/redis/redis.service';

/**
 * Defines the CarModule, which encapsulates all car-related functionality.
 * 
 * This module imports necessary dependencies, defines controllers and providers,
 * and exports the CarService for use in other modules.
 * 
 * @remarks
 * The module uses MongooseModule for database operations, and imports various
 * related modules such as FilterModule, UserModule, AuthenticationModule, 
 * AlertPriceModule, and HttpModule.
 * 
 * @param imports - An array of modules to import, including database configurations
 *                  and related feature modules.
 * @param controllers - An array specifying the controllers to be instantiated
 *                      within this module.
 * @param providers - An array of providers to be available within this module.
 * @param exports - An array of providers to export from this module, making them
 *                  available for use in other modules.
 * 
 * @returns A NestJS module definition for car-related functionality.
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: Car.name, schema: CarSchema }]),
    // ClientsModule.register([
    //   {
    //     name: 'QUEUE_ADVERTISEMENT_SERVICE',
    //     transport: Transport.RMQ,
    //     options: {
    //       urls: [config.RABBITMQ_CAR_URL],
    //       queue: config.QUEUE_ADVERTISEMENT_QUEUE_NAME,
    //       // prefetchCount: 1,
    //       queueOptions: {
    //         durable: true
    //       },
    //     },
    //   },
    // ]),
    forwardRef(() => FilterModule),
    forwardRef(() => UserModule),
    AuthenticationModule,
    forwardRef(() => AlertPriceModule),
    HttpModule
  ],
  controllers: [CarController],
  providers: [CarService, AwsS3Service, CarConsumerService, MailService, RedisService],
  exports: [CarService]
})
export class CarModule {}
