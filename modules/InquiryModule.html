import { Module } from '@nestjs/common';
import { InquiryService } from './inquiry.service';
import { InquiryController } from './inquiry.controller';

// mongoose
import { Inquiry, InquirySchema } from './schema/inquiry.schema';
import { MongooseModule } from '@nestjs/mongoose';

// user role & permission
import { AuthenticationModule } from '../authentication/authentication.module';
import { UserModule } from '../user/user.module';
import { CarModule } from '../car/car.module';

// send mail for inquiry & appoitment
import { MailService } from '../../utilities/services/mail/mail.service';

/**
 * Defines the Inquiry module for handling inquiries in the application.
 * 
 * @decorator @Module
 * @description This module sets up the necessary imports, controllers, and providers
 *              for managing inquiries, including database integration, authentication,
 *              user management, car-related functionality, and email services.
 * 
 * @property {Array} imports - An array of modules to import:
 *   - MongooseModule.forFeature: Registers the Inquiry schema with Mongoose.
 *   - UserModule: Provides user-related functionality.
 *   - AuthenticationModule: Handles authentication and authorization.
 *   - CarModule: Manages car-related operations.
 * 
 * @property {Array} controllers - An array containing the InquiryController.
 * 
 * @property {Array} providers - An array of service providers:
 *   - InquiryService: Handles inquiry-related business logic.
 *   - MailService: Provides email functionality for inquiries and appointments.
 * 
 * @returns {Object} The configured Inquiry module.
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: Inquiry.name, schema: InquirySchema }]),
    UserModule,
    AuthenticationModule,
    CarModule
  ],
  controllers: [InquiryController],
  providers: [InquiryService, MailService],
})
export class InquiryModule {}
