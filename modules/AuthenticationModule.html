Description

Defines the Authentication module for the application. This module handles user authentication and related functionality.

  • MongooseModule.forFeature: Imports the Authentication schema for database operations.
  • forwardRef(() => UserModule): Imports the UserModule with a forward reference to avoid circular dependencies.
  • AuthenticationController: Handles authentication-related HTTP requests.
  • AuthenticationService: Provides authentication-related business logic.
  • MailService: Provides email sending functionality.
  • AuthenticationService: Makes the AuthenticationService available for use in other modules.

import { Module, forwardRef } from '@nestjs/common';
import { AuthenticationService } from './authentication.service';
import { AuthenticationController } from './authentication.controller';

// mongoose
import { MongooseModule } from '@nestjs/mongoose';
import { Authentication, AuthenticationSchema } from './schema/authentication.schema';
import { UserModule } from '../user/user.module';

// send mail service
import { MailService } from '../../utilities/services/mail/mail.service';

/**
 * Defines the Authentication module for the application.
 * This module handles user authentication and related functionality.
 *
 * @decorator @Module
 * @param {Object} options - The module configuration options.
 * @param {Array} options.imports - The list of imported modules.
 *   - MongooseModule.forFeature: Imports the Authentication schema for database operations.
 *   - forwardRef(() => UserModule): Imports the UserModule with a forward reference to avoid circular dependencies.
 * @param {Array} options.controllers - The list of controllers in this module.
 *   - AuthenticationController: Handles authentication-related HTTP requests.
 * @param {Array} options.providers - The list of providers (services) in this module.
 *   - AuthenticationService: Provides authentication-related business logic.
 *   - MailService: Provides email sending functionality.
 * @param {Array} options.exports - The list of providers exported from this module.
 *   - AuthenticationService: Makes the AuthenticationService available for use in other modules.
 * @returns {void}
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: Authentication.name, schema: AuthenticationSchema }]), 
    forwardRef(() => UserModule)
  ],
  controllers: [AuthenticationController],
  providers: [AuthenticationService, MailService],
  exports: [AuthenticationService]
})
export class AuthenticationModule { }
