Description

Defines the AlertSubscribeModule, which manages alert subscriptions in the application.

This module integrates various dependencies and services required for alert subscription functionality, including database access, authentication, filtering, and email services.

The module uses MongooseModule for database operations, and includes several other modules for user management, authentication, filtering, and car-related operations.

import { Module } from '@nestjs/common';
import { AlertSubscribeService } from './alert-subscribe.service';
import { AlertSubscribeController } from './alert-subscribe.controller';

// mongoose
import { MongooseModule } from '@nestjs/mongoose';
import { AlertSubscribe, AlertSubscribeSchema } from './schema/alert-subscribe.schema';

// permission & authentication
import { UserModule } from '../user/user.module';
import { AuthenticationModule } from '../authentication/authentication.module';

// mail service
import { MailService } from '../../utilities/services/mail/mail.service';

// filter module
import { FilterModule } from '../filter/filter.module';

// car module for mail
import { CarModule } from '../car/car.module';

/**
 * Defines the AlertSubscribeModule, which manages alert subscriptions in the application.
 * 
 * This module integrates various dependencies and services required for alert subscription functionality,
 * including database access, authentication, filtering, and email services.
 * 
 * @remarks
 * The module uses MongooseModule for database operations, and includes several other modules
 * for user management, authentication, filtering, and car-related operations.
 * 
 * @module AlertSubscribeModule
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: AlertSubscribe.name, schema: AlertSubscribeSchema }]),
    UserModule,
    AuthenticationModule,
    FilterModule,
    CarModule
  ],
  controllers: [AlertSubscribeController],
  providers: [AlertSubscribeService, MailService],
})
export class AlertSubscribeModule { }
